231 PluginReceiveResult NatPunchthroughServer::OnReceive(Packet *packet)
232 {
233  switch (packet->data[0])
234  {
235  case ID_NAT_PUNCHTHROUGH_REQUEST:
236  OnNATPunchthroughRequest(packet);
237  return RR_STOP_PROCESSING_AND_DEALLOCATE;
238  case ID_NAT_GET_MOST_RECENT_PORT:
239  OnGetMostRecentPort(packet);
240  return RR_STOP_PROCESSING_AND_DEALLOCATE;
241  case ID_NAT_CLIENT_READY:
242  OnClientReady(packet);
243  return RR_STOP_PROCESSING_AND_DEALLOCATE;
244  case ID_NAT_REQUEST_BOUND_ADDRESSES:
245  {
246  SLNet::BitStream outgoingBs;
247  outgoingBs.Write((MessageID)ID_NAT_RESPOND_BOUND_ADDRESSES);
248 
249  if (boundAddresses[0]==UNASSIGNED_SYSTEM_ADDRESS)
250  {
251  DataStructures::List<RakNetSocket2* > sockets;
252  rakPeerInterface->GetSockets(sockets);
253  for (unsigned i=0; i < sockets.Size() && i < MAXIMUM_NUMBER_OF_INTERNAL_IDS; i++)
254  {
255  boundAddresses[i]=sockets[i]->GetBoundAddress();
256  boundAddressCount++;
257  }
258  }
259 
260  outgoingBs.Write(boundAddressCount);
261  for (int i=0; i < boundAddressCount; i++)
262  {
263  outgoingBs.Write(boundAddresses[i]);
264  }
265 
266  rakPeerInterface->Send(&outgoingBs,HIGH_PRIORITY,RELIABLE_ORDERED,0,packet->systemAddress,false);
267  }
268  return RR_STOP_PROCESSING_AND_DEALLOCATE;
269  case ID_NAT_PING:
270  {
271  }
272  return RR_STOP_PROCESSING_AND_DEALLOCATE;
273  case ID_OUT_OF_BAND_INTERNAL:
274  if (packet->length>=2 && packet->data[1]==ID_NAT_PING)
275  {
276  SLNet::BitStream bs(packet->data,packet->length,false);
277  bs.IgnoreBytes(sizeof(MessageID)*2);
278  uint16_t externalPort;
279  bs.Read(externalPort);
280 
281  SLNet::BitStream outgoingBs;
282  outgoingBs.Write((MessageID)ID_NAT_PONG);
283  outgoingBs.Write(externalPort);
284  uint16_t externalPort2 = packet->systemAddress.GetPort();
285  outgoingBs.Write(externalPort2);
286  rakPeerInterface->SendOutOfBand((const char*) packet->systemAddress.ToString(false),packet->systemAddress.GetPort(),(const char*) outgoingBs.GetData(),outgoingBs.GetNumberOfBytesUsed());
287 
288  return RR_STOP_PROCESSING_AND_DEALLOCATE;
289  }
290  }
291  return RR_CONTINUE_PROCESSING;

386 void NatPunchthroughServer::OnNATPunchthroughRequest(Packet *packet)
387 {
388  SLNet::BitStream outgoingBs;
389  SLNet::BitStream incomingBs(packet->data, packet->length, false);
390  incomingBs.IgnoreBytes(sizeof(MessageID));
391  RakNetGUID recipientGuid, senderGuid;
392  incomingBs.Read(recipientGuid);
393  senderGuid=packet->guid;
394  unsigned int i;
395  bool objectExists;
396  i = users.GetIndexFromKey(senderGuid, &objectExists);
397  RakAssert(objectExists);
398 
399  ConnectionAttempt *ca = SLNet::OP_NEW<ConnectionAttempt>(_FILE_AND_LINE_);
400  ca->sender=users[i];
401  ca->sessionId=sessionId++;
402  i = users.GetIndexFromKey(recipientGuid, &objectExists);
403  if (objectExists==false || ca->sender == ca->recipient)
404  {
405 // printf("DEBUG %i\n", __LINE__);
406 // printf("DEBUG recipientGuid=%s\n", recipientGuid.ToString());
407 // printf("DEBUG users[0] guid=%s\n", users[0]->guid.ToString());
408 
409  outgoingBs.Write((MessageID)ID_NAT_TARGET_NOT_CONNECTED);
410  outgoingBs.Write(recipientGuid);
411  rakPeerInterface->Send(&outgoingBs,HIGH_PRIORITY,RELIABLE_ORDERED,0,packet->systemAddress,false);
412  SLNet::OP_DELETE(ca,_FILE_AND_LINE_);
413  return;
414  }
415  ca->recipient=users[i];
416  if (ca->recipient->HasConnectionAttemptToUser(ca->sender))
417  {
418  outgoingBs.Write((MessageID)ID_NAT_ALREADY_IN_PROGRESS);
419  outgoingBs.Write(recipientGuid);
420  rakPeerInterface->Send(&outgoingBs,HIGH_PRIORITY,RELIABLE_ORDERED,0,packet->systemAddress,false);
421  SLNet::OP_DELETE(ca,_FILE_AND_LINE_);
422  return;
423  }
424 
425  ca->sender->connectionAttempts.Insert(ca, _FILE_AND_LINE_ );
426  ca->recipient->connectionAttempts.Insert(ca, _FILE_AND_LINE_ );
427 
428  StartPunchthroughForUser(ca->sender);
429 }

442 {

444  bsIn.IgnoreBytes(sizeof(MessageID));
445  uint16_t curSessionId;
446  unsigned short mostRecentPort;
447  bsIn.Read(curSessionId);
448  bsIn.Read(mostRecentPort);
449 
450  unsigned int i,j;
451  User *user;
452  ConnectionAttempt *connectionAttempt;
453  bool objectExists;
454  i = users.GetIndexFromKey(packet->guid, &objectExists);
455 
456  if (natPunchthroughServerDebugInterface)

458  SLNet::RakString log;

462  log= SLNet::RakString("Got ID_NAT_GET_MOST_RECENT_PORT from systemAddress %s guid %s. port=%i. sessionId=%i. userFound=%i.", addr1, addr2, mostRecentPort, curSessionId, objectExists);
463  natPunchthroughServerDebugInterface->OnServerMessage(log.C_String());
464  }
465 
466  if (objectExists)
467  {
468  user=users[i];
469  user->mostRecentPort=mostRecentPort;
470  SLNet::Time time = SLNet::GetTime();
471 
472  for (j=0; j < user->connectionAttempts.Size(); j++)
473  {
474  connectionAttempt=user->connectionAttempts[j];
475  if (connectionAttempt->attemptPhase==ConnectionAttempt::NAT_ATTEMPT_PHASE_GETTING_RECENT_PORTS &&
476  connectionAttempt->sender->mostRecentPort!=0 &&
477  connectionAttempt->recipient->mostRecentPort!=0 &&
478  // 04/29/08 add sessionId to prevent processing for other systems
479  connectionAttempt->sessionId== curSessionId)
480  {
481  SystemAddress senderSystemAddress = connectionAttempt->sender->systemAddress;
482  SystemAddress recipientSystemAddress = connectionAttempt->recipient->systemAddress;
483  SystemAddress recipientTargetAddress = recipientSystemAddress;
484  SystemAddress senderTargetAddress = senderSystemAddress;

486  senderTargetAddress.SetPortHostOrder(connectionAttempt->sender->mostRecentPort);
487 
488  // Pick a time far enough in the future that both systems will have gotten the message
489  int targetPing = rakPeerInterface->GetAveragePing(recipientTargetAddress);
490  int senderPing = rakPeerInterface->GetAveragePing(senderSystemAddress);

493  simultaneousAttemptTime = time + 1500;
494  else

497  if (largerPing * 4 < 100)

522  bsOut.Write(connectionAttempt->sender->guid);

578

581  unsigned int i;
582  for (i=0; i < user->connectionAttempts.Size(); i++)
583  {
584  connectionAttempt=user->connectionAttempts[i];
585  if (connectionAttempt->sender==user)
586  {
