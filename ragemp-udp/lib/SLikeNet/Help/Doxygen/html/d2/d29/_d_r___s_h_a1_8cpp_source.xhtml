119 void CSHA1::Update(const UINT_8* pbData, UINT_32 uLen)
120 {
121  UINT_32 j = ((m_count[0] >> 3) & 0x3F);
122 
123  if((m_count[0] += (uLen << 3)) < (uLen << 3))
124  ++m_count[1]; // Overflow
125 
126  m_count[1] += (uLen >> 29);
127 
128  UINT_32 i;
129  if((j + uLen) > 63)
130  {
131  i = 64 - j;
132  memcpy(&m_buffer[j], pbData, i);
133  Transform(m_state, m_buffer);
134 
135  for( ; (i + 63) < uLen; i += 64)
136  Transform(m_state, &pbData[i]);
137 
138  j = 0;
139  }
140  else i = 0;
141 
142  if((uLen - i) != 0)
143  memcpy(&m_buffer[j], &pbData[i], uLen - i);
144 }

178 void CSHA1::Final()
179 {
180  UINT_32 i;
181 
182  UINT_8 pbFinalCount[8];
183  for(i = 0; i < 8; ++i)
184  pbFinalCount[i] = static_cast<UINT_8>((m_count[((i >= 4) ? 0 : 1)] >>
185  ((3 - (i & 3)) * 8) ) & 0xFF); // Endian independent
186 
187  Update((UINT_8*)"\200", 1);
188 
189  while((m_count[0] & 504) != 448)
190  Update((UINT_8*)"\0", 1);
191 
192  Update(pbFinalCount, 8); // Cause a Transform()
193 
194  for(i = 0; i < 20; ++i)
195  m_digest[i] = static_cast<UINT_8>((m_state[i >> 2] >> ((3 -
196  (i & 3)) * 8)) & 0xFF);
197 
198  // Wipe variables for security reasons
199 #ifdef SHA1_WIPE_VARIABLES
200  memset(m_buffer, 0, 64);
201  memset(m_state, 0, 20);
202  memset(m_count, 0, 8);
203  memset(pbFinalCount, 0, 8);
204  Transform(m_state, m_buffer);
205 #endif
206 }

269 // Sample code: http://www.opensource.apple.com/source/freeradius/freeradius-11/freeradius/src/lib/hmac.c
270 void CSHA1::HMAC(unsigned char *sharedKey, int sharedKeyLength, unsigned char *data, int dataLength, unsigned char output[SHA1_LENGTH])
271 {
272  // 1. Append zeros to the end of K to create a 64 byte string
273  static const int sha1BlockLength=64;
274 
275  if (sharedKeyLength > sha1BlockLength)
276  sharedKeyLength = sha1BlockLength;
277 
278  // ipad = the byte 0x36 repeated 64 times
279  // opad = the byte 0x5C repeated 64 times
280  unsigned char keyWithIpad[sha1BlockLength];
281  unsigned char keyWithOpad[sha1BlockLength];
282 
283  memset( keyWithIpad, 0, sizeof(keyWithIpad));
284  memset( keyWithOpad, 0, sizeof(keyWithOpad));
285  memcpy( keyWithIpad, sharedKey, sharedKeyLength);
286  memcpy( keyWithOpad, sharedKey, sharedKeyLength);
287 
288  for (int i = 0; i < sha1BlockLength; i++) {
289  keyWithIpad[i] ^= 0x36;
290  keyWithOpad[i] ^= 0x5c;
291  }
292 
293  // 3. Append the data stream Text to the 64 byte string resulting from step (2)
294  // 4. Apply H to the stream generated in step (3)
295  CSHA1 firstHash;
296  firstHash.Reset();
297  firstHash.Update( keyWithIpad, sha1BlockLength );
298  firstHash.Update( data, dataLength );
299  firstHash.Final();
300 
301  // 6. Append the H (hash) result from step (4) to the 64 byte string resulting from step (5)
302  // 7. Apply H to the stream generated in step (6) and output the result
303  CSHA1 secondHash;
304  secondHash.Reset();
305  secondHash.Update( keyWithOpad, sha1BlockLength );
306  secondHash.Update( firstHash.GetHash(), SHA1_LENGTH );
307  secondHash.Final();
308 
309  memcpy(output, secondHash.GetHash(), SHA1_LENGTH);
