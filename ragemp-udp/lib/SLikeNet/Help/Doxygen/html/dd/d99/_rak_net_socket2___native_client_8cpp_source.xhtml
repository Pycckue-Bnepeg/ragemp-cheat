97 void RNS2_NativeClient::onRecvFrom(void* pData, int32_t dataSize)
98 {
99  RNS2RecvStruct *recvStruct = (RNS2RecvStruct *) pData;
100  RNS2_NativeClient *socket2 = (RNS2_NativeClient *) recvStruct->socket;
101 
102  //any error codes will be given to us in the dataSize value; see pp_errors.h for a list of errors
103  if(dataSize <=0 || !pData )
104  {
105  // Free data
106  socket2->eventHandler->DeallocRNS2RecvStruct(recvStruct, _FILE_AND_LINE_);
107 
108  // This value indicates failure due to an asynchronous operation being
109  // interrupted. The most common cause of this error code is destroying
110  // a resource that still has a callback pending. All callbacks are
111  // guaranteed to execute, so any callbacks pending on a destroyed
112  // resource will be issued with PP_ERROR_ABORTED.
113  if(dataSize==PP_ERROR_ABORTED)
114  {
115  RAKNET_DEBUG_PRINTF("onRecvFrom error PP_ERROR_ABORTED, killing recvfrom loop\n");
116  }
117  else
118  {
119  RAKNET_DEBUG_PRINTF("onRecvFrom error %d\n", dataSize);
120 
121  // Reissue call
122  socket2->IssueReceiveCall();
123  }
124 
125  return;
126  }
127 
128  recvStruct->bytesRead=dataSize;
129  recvStruct->timeRead= SLNet::GetTimeUS();
130 
131 
132  PP_NetAddress_Private addr;
133  bool ok=false;
134  if(((PPB_UDPSocket_Private_0_4*) Module::Get()->GetBrowserInterface(PPB_UDPSOCKET_PRIVATE_INTERFACE_0_4))->GetRecvFromAddress(socket2->rns2Socket, &addr) == PP_TRUE)
135  {
136  PP_NetAddressFamily_Private family = NetAddressPrivate::GetFamily(addr);
137  if (family == PP_NETADDRESSFAMILY_IPV4)
138  {
139  ok = NetAddressPrivate::GetAddress(addr, &recvStruct->systemAddress.address.addr4.sin_addr, sizeof(in_addr));
140  }
141 #if RAKNET_SUPPORT_IPV6==1
142  else
143  {
144  ok = NetAddressPrivate::GetAddress(addr, &recvStruct->systemAddress.address.addr6.sin6_addr, sizeof(in6_addr));
145  }
146 #endif
147  }
148 
149  if(ok)
150  {
151  recvStruct->systemAddress.SetPortHostOrder(pp::NetAddressPrivate::GetPort(addr));
152  socket2->binding.eventHandler->OnRNS2Recv(recvStruct);
153  }
154 
155  // Reissue call
156  socket2->IssueReceiveCall();
157 }
