80 void NatTypeDetectionClient::OnCompletion(NATTypeDetectionResult result)
81 {
82  Packet *p = AllocatePacketUnified(sizeof(MessageID)+sizeof(unsigned char)*2);
83  //printf("Returning nat detection result to the user\n");
84  p->data[0]=ID_NAT_TYPE_DETECTION_RESULT;
85  p->systemAddress=serverAddress;
86  p->systemAddress.systemIndex=(SystemIndex)-1;
87  p->guid=rakPeerInterface->GetGuidFromSystemAddress(serverAddress);
88  p->data[1]=(unsigned char) result;
89  p->wasGeneratedLocally=true;
90  rakPeerInterface->PushBackPacket(p, true);
91 
92  // Symmetric and port restricted are determined by server, so no need to notify server we are done
93  if (result!=NAT_TYPE_PORT_RESTRICTED && result!=NAT_TYPE_SYMMETRIC)
94  {
95  // Otherwise tell the server we got this message, so it stops sending tests to us
96  SLNet::BitStream bs;
97  bs.Write((unsigned char)ID_NAT_TYPE_DETECTION_REQUEST);
98  bs.Write(false); // Done
99  rakPeerInterface->Send(&bs,HIGH_PRIORITY,RELIABLE,0,serverAddress,false);
100  }
101 
102  Shutdown();
103 }

108 void NatTypeDetectionClient::Update(void)
109 {
110  if (IsInProgress())
111  {
112  RNS2RecvStruct *recvStruct;
113  bufferedPacketsMutex.Lock();
114  if (bufferedPackets.Size()>0)
115  recvStruct=bufferedPackets.Pop();
116  else
117  recvStruct=0;
118  bufferedPacketsMutex.Unlock();
119  while (recvStruct)
120  {
121  if (recvStruct->bytesRead==1 && recvStruct->data[0]==NAT_TYPE_NONE)
122  {
123  OnCompletion(NAT_TYPE_NONE);
124  RakAssert(IsInProgress()==false);
125  }
126  DeallocRNS2RecvStruct(recvStruct, _FILE_AND_LINE_);
127 
128  bufferedPacketsMutex.Lock();
129  if (bufferedPackets.Size()>0)
130  recvStruct=bufferedPackets.Pop();
131  else
132  recvStruct=0;
133  bufferedPacketsMutex.Unlock();
134  }
135  }

137 PluginReceiveResult NatTypeDetectionClient::OnReceive(Packet *packet)
138 {
139  if (IsInProgress())
140  {
141  switch (packet->data[0])
142  {
143  case ID_OUT_OF_BAND_INTERNAL:
144  {
145  if (packet->length>=3 && packet->data[1]==ID_NAT_TYPE_DETECT)
146  {
147  OnCompletion((NATTypeDetectionResult)packet->data[2]);
148  return RR_STOP_PROCESSING_AND_DEALLOCATE;
149  }
150  }
151  break;
152  case ID_NAT_TYPE_DETECTION_RESULT:
153  if (packet->wasGeneratedLocally==false)
154  {
155  OnCompletion((NATTypeDetectionResult)packet->data[1]);
156  return RR_STOP_PROCESSING_AND_DEALLOCATE;
157  }
158  else
159  break;
160  case ID_NAT_TYPE_DETECTION_REQUEST:
161  OnTestPortRestricted(packet);
162  return RR_STOP_PROCESSING_AND_DEALLOCATE;
163  }
164  }
165 
166  return RR_CONTINUE_PROCESSING;
167 }

184 void NatTypeDetectionClient::OnTestPortRestricted(Packet *packet)
185 {
186  SLNet::BitStream bsIn(packet->data,packet->length,false);
187  bsIn.IgnoreBytes(sizeof(MessageID));
188  SLNet::RakString s3p4StrAddress;
189  bsIn.Read(s3p4StrAddress);
190  unsigned short s3p4Port;
191  bsIn.Read(s3p4Port);
192 
193  DataStructures::List<RakNetSocket2* > sockets;
194  rakPeerInterface->GetSockets(sockets);
195  SystemAddress s3p4Addr = sockets[0]->GetBoundAddress();
196  s3p4Addr.FromStringExplicitPort(s3p4StrAddress.C_String(), s3p4Port);
197 
198  // Send off the RakNet socket to the specified address, message is unformatted
199  // Server does this twice, so don't have to unduly worry about packetloss
200  SLNet::BitStream bsOut;
201  bsOut.Write((MessageID) NAT_TYPE_PORT_RESTRICTED);
202  bsOut.Write(rakPeerInterface->GetGuidFromSystemAddress(UNASSIGNED_SYSTEM_ADDRESS));
203 // SocketLayer::SendTo_PC( sockets[0], (const char*) bsOut.GetData(), bsOut.GetNumberOfBytesUsed(), s3p4Addr, __FILE__, __LINE__ );
204 
205  RNS2_SendParameters bsp;
206  bsp.data = (char*) bsOut.GetData();
207  bsp.length = bsOut.GetNumberOfBytesUsed();
208  bsp.systemAddress=s3p4Addr;
209  sockets[0]->Send(&bsp, _FILE_AND_LINE_);
210 
211 }
212 void NatTypeDetectionClient::Shutdown(void)
213 {
214  serverAddress=UNASSIGNED_SYSTEM_ADDRESS;
215  if (c2!=0)
216  {
217 #if !defined(__native_client__) && !defined(WINDOWS_STORE_RT)

221 
222  SLNet::OP_DELETE(c2, _FILE_AND_LINE_);
223  c2=0;
224  }

230 }
