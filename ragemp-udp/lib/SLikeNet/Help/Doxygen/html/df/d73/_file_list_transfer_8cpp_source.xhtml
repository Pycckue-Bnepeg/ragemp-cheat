281 bool FileListTransfer::DecodeSetHeader(Packet *packet)
282 {
283  bool anythingToWrite=false;
284  unsigned short setID;
285  SLNet::BitStream inBitStream(packet->data, packet->length, false);
286  inBitStream.IgnoreBits(8);
287  inBitStream.Read(setID);
288  FileListReceiver *fileListReceiver;
289  if (fileListReceivers.Has(setID)==false)
290  {
291  // If this assert hits you didn't call SetupReceive
292 #ifdef _DEBUG
293  RakAssert(0);
294 #endif
295  return false;
296  }
297  fileListReceiver=fileListReceivers.Get(setID);
298  if (fileListReceiver->allowedSender!=packet->systemAddress)
299  {
300 #ifdef _DEBUG
301  RakAssert(0);
302 #endif
303  return false;
304  }
305 
306 #ifdef _DEBUG
307  RakAssert(fileListReceiver->gotSetHeader==false);
308 #endif
309 
310  inBitStream.Read(anythingToWrite);
311 
312  if (anythingToWrite)
313  {
314  inBitStream.ReadCompressed(fileListReceiver->setCount);
315  if (inBitStream.ReadCompressed(fileListReceiver->setTotalFinalLength))
316  {
317  fileListReceiver->setTotalCompressedTransmissionLength=fileListReceiver->setTotalFinalLength;
318  fileListReceiver->gotSetHeader=true;
319  return true;
320  }
321 
322  }
323  else
324  {
325  FileListTransferCBInterface::DownloadCompleteStruct dcs;
326  dcs.setID=fileListReceiver->setID;
327  dcs.numberOfFilesInThisSet=fileListReceiver->setCount;
328  dcs.byteLengthOfThisSet=fileListReceiver->setTotalFinalLength;
329  dcs.senderSystemAddress=packet->systemAddress;
330  dcs.senderGuid=packet->guid;
331 
332  if (fileListReceiver->downloadHandler->OnDownloadComplete(&dcs)==false)
333  {
334  fileListReceiver->downloadHandler->OnDereference();
335  fileListReceivers.Delete(setID);
336  if (fileListReceiver->deleteDownloadHandler)
337  SLNet::OP_DELETE(fileListReceiver->downloadHandler, _FILE_AND_LINE_);
338  SLNet::OP_DELETE(fileListReceiver, _FILE_AND_LINE_);
339  }
340 
341  return true;
342  }
343 
344  return false;
345 }

347 bool FileListTransfer::DecodeFile(Packet *packet, bool isTheFullFile)
348 {
349  FileListTransferCBInterface::OnFileStruct onFileStruct;
350  SLNet::BitStream inBitStream(packet->data, packet->length, false);
351  inBitStream.IgnoreBits(8);
352 
353  onFileStruct.senderSystemAddress=packet->systemAddress;
354  onFileStruct.senderGuid=packet->guid;
355 
356  unsigned int partCount=0;
357  unsigned int partTotal=0;
358  unsigned int partLength=0;
359  onFileStruct.fileData=0;
360  if (isTheFullFile==false)
361  {
362  // Disable endian swapping on reading this, as it's generated locally in ReliabilityLayer.cpp
363  inBitStream.ReadBits( (unsigned char* ) &partCount, BYTES_TO_BITS(sizeof(partCount)), true );
364  inBitStream.ReadBits( (unsigned char* ) &partTotal, BYTES_TO_BITS(sizeof(partTotal)), true );
365  inBitStream.ReadBits( (unsigned char* ) &partLength, BYTES_TO_BITS(sizeof(partLength)), true );
366  inBitStream.IgnoreBits(8);
367  // The header is appended to every chunk, which we continue to read after this statement flrMemoryBlock
368  }
369  inBitStream >> onFileStruct.context;
370  // inBitStream.Read(onFileStruct.context);
371  inBitStream.Read(onFileStruct.setID);
372  FileListReceiver *fileListReceiver;
373  if (fileListReceivers.Has(onFileStruct.setID)==false)
374  {
375  return false;
376  }
377  fileListReceiver=fileListReceivers.Get(onFileStruct.setID);
378  if (fileListReceiver->allowedSender!=packet->systemAddress)
379  {
380 #ifdef _DEBUG
381  RakAssert(0);
382 #endif
383  return false;
384  }
385 
386 #ifdef _DEBUG
387  RakAssert(fileListReceiver->gotSetHeader==true);
388 #endif
389 
390  if (StringCompressor::Instance()->DecodeString(onFileStruct.fileName, 512, &inBitStream)==false)
391  {
392 #ifdef _DEBUG
393  RakAssert(0);
394 #endif
395  return false;
396  }
397 
398  inBitStream.ReadCompressed(onFileStruct.fileIndex);
399  inBitStream.ReadCompressed(onFileStruct.byteLengthOfThisFile);
400 
401  onFileStruct.numberOfFilesInThisSet=fileListReceiver->setCount;
402  onFileStruct.byteLengthOfThisSet=fileListReceiver->setTotalFinalLength;
403 
404  if (isTheFullFile)
405  {
406  onFileStruct.bytesDownloadedForThisFile=onFileStruct.byteLengthOfThisFile;
407  fileListReceiver->setTotalDownloadedLength+=onFileStruct.byteLengthOfThisFile;
408  onFileStruct.bytesDownloadedForThisSet=fileListReceiver->setTotalDownloadedLength;
409  }
410  else
411  {
412  onFileStruct.bytesDownloadedForThisFile=partLength*partCount;
413  onFileStruct.bytesDownloadedForThisSet=fileListReceiver->setTotalDownloadedLength+onFileStruct.bytesDownloadedForThisFile;
414  }
415 
416  // User callback for this file.
417  if (isTheFullFile)
418  {
419  inBitStream.AlignReadToByteBoundary();
420  onFileStruct.fileData = (char*) rakMalloc_Ex( (size_t) onFileStruct.byteLengthOfThisFile, _FILE_AND_LINE_ );
421  inBitStream.Read((char*)onFileStruct.fileData, onFileStruct.byteLengthOfThisFile);
422 
423  FileListTransferCBInterface::FileProgressStruct fps;
424  fps.onFileStruct=&onFileStruct;
425  fps.partCount=1;
426  fps.partTotal=1;
427  fps.dataChunkLength=onFileStruct.byteLengthOfThisFile;
428  fps.firstDataChunk=onFileStruct.fileData;
429  fps.iriDataChunk=onFileStruct.fileData;
430  fps.allocateIrIDataChunkAutomatically=true;
431  fps.iriWriteOffset=0;
432  fps.senderSystemAddress=packet->systemAddress;
433  fps.senderGuid=packet->guid;
434  fileListReceiver->downloadHandler->OnFileProgress(&fps);
435 
436  // Got a complete file
437  // Either we are using IncrementalReadInterface and it was a small file or
438  // We are not using IncrementalReadInterface
439  if (fileListReceiver->downloadHandler->OnFile(&onFileStruct))
440  rakFree_Ex(onFileStruct.fileData, _FILE_AND_LINE_ );
441 
442  fileListReceiver->filesReceived++;
443 
444  // If this set is done, free the memory for it.
445  if ((int) fileListReceiver->setCount==fileListReceiver->filesReceived)
446  {
447  FileListTransferCBInterface::DownloadCompleteStruct dcs;
448  dcs.setID=fileListReceiver->setID;
449  dcs.numberOfFilesInThisSet=fileListReceiver->setCount;
450  dcs.byteLengthOfThisSet=fileListReceiver->setTotalFinalLength;
451  dcs.senderSystemAddress=packet->systemAddress;
452  dcs.senderGuid=packet->guid;
453 
454  if (fileListReceiver->downloadHandler->OnDownloadComplete(&dcs)==false)
455  {
456  fileListReceiver->downloadHandler->OnDereference();
457  if (fileListReceiver->deleteDownloadHandler)
458  SLNet::OP_DELETE(fileListReceiver->downloadHandler, _FILE_AND_LINE_);
459  fileListReceivers.Delete(onFileStruct.setID);
460  SLNet::OP_DELETE(fileListReceiver, _FILE_AND_LINE_);
461  }
462  }
463 
464  }
465  else
466  {
467  inBitStream.AlignReadToByteBoundary();
468 
469  char *firstDataChunk;
470  unsigned int unreadBits = inBitStream.GetNumberOfUnreadBits();
471  unsigned int unreadBytes = BITS_TO_BYTES(unreadBits);
472  firstDataChunk=(char*) inBitStream.GetData()+BITS_TO_BYTES(inBitStream.GetReadOffset());
473 
474  FileListTransferCBInterface::FileProgressStruct fps;
475  fps.onFileStruct=&onFileStruct;
476  fps.partCount=partCount;
477  fps.partTotal=partTotal;
478  fps.dataChunkLength=unreadBytes;
479  fps.firstDataChunk=firstDataChunk;
480  fps.iriDataChunk=0;
481  fps.allocateIrIDataChunkAutomatically=true;
482  fps.iriWriteOffset=0;
483  fps.senderSystemAddress=packet->systemAddress;
484  fps.senderGuid=packet->guid;
485 
486  // Remote system is sending a complete file, but the file is large enough that we get ID_PROGRESS_NOTIFICATION from the transport layer
487  fileListReceiver->downloadHandler->OnFileProgress(&fps);
488 
489  }
490 
491  return true;

493 PluginReceiveResult FileListTransfer::OnReceive(Packet *packet)
494 {
495  switch (packet->data[0])
496  {
497  case ID_FILE_LIST_TRANSFER_HEADER:
498  DecodeSetHeader(packet);
499  return RR_STOP_PROCESSING_AND_DEALLOCATE;
500  case ID_FILE_LIST_TRANSFER_FILE:
501  DecodeFile(packet, true);
502  return RR_STOP_PROCESSING_AND_DEALLOCATE;
503  case ID_FILE_LIST_REFERENCE_PUSH:
504  OnReferencePush(packet, true);
505  return RR_STOP_PROCESSING_AND_DEALLOCATE;
506  case ID_FILE_LIST_REFERENCE_PUSH_ACK:
507  OnReferencePushAck(packet);
508  return RR_STOP_PROCESSING_AND_DEALLOCATE;
509  case ID_DOWNLOAD_PROGRESS:
510  if (packet->length>sizeof(MessageID)+sizeof(unsigned int)*3)
511  {
512  if (packet->data[sizeof(MessageID)+sizeof(unsigned int)*3]==ID_FILE_LIST_TRANSFER_FILE)
513  {
514  DecodeFile(packet, false);
515  return RR_STOP_PROCESSING_AND_DEALLOCATE;
516  }
517  if (packet->data[sizeof(MessageID)+sizeof(unsigned int)*3]==ID_FILE_LIST_REFERENCE_PUSH)
518  {
519  OnReferencePush(packet, false);
520  return RR_STOP_PROCESSING_AND_DEALLOCATE;
521  }
522  }
523  break;
524  }
525 
526  return RR_CONTINUE_PROCESSING;

684  {

696  unsigned int partLength=0;

705  // The header is appended to every chunk, which we continue to read after this statement flrMemoryBlock
706  }

708  inBitStream >> onFileStruct.context;
709  inBitStream.Read(onFileStruct.setID);
710 
711  // This is not a progress notification, it is actually the entire packet
712  if (isTheFullFile==true)
713  {
714  refPushAck.Write((MessageID)ID_FILE_LIST_REFERENCE_PUSH_ACK);
715  refPushAck.Write(onFileStruct.setID);
716  SendUnified(&refPushAck,HIGH_PRIORITY, RELIABLE, 0, packet->systemAddress, false);
717  }
718 
719  // inBitStream.Read(onFileStruct.context);
720  FileListReceiver *fileListReceiver;
721  if (fileListReceivers.Has(onFileStruct.setID)==false)
722  {
723  return;
724  }
725  fileListReceiver=fileListReceivers.Get(onFileStruct.setID);
726  if (fileListReceiver->allowedSender!=packet->systemAddress)
727  {
728 #ifdef _DEBUG
729  RakAssert(0);
730 #endif
731  return;
732  }
733 
734 #ifdef _DEBUG
735  RakAssert(fileListReceiver->gotSetHeader==true);
736 #endif
737 
738  if (StringCompressor::Instance()->DecodeString(onFileStruct.fileName, 512, &inBitStream)==false)
739  {
740 #ifdef _DEBUG
741  RakAssert(0);
742 #endif
743  return;
744  }
745 
746  inBitStream.ReadCompressed(onFileStruct.fileIndex);
747  inBitStream.ReadCompressed(onFileStruct.byteLengthOfThisFile);

751  inBitStream.ReadCompressed(chunkLength);
752 
753  bool lastChunk=false;
754  inBitStream.Read(lastChunk);
755  bool finished = lastChunk && isTheFullFile;
756 
757  if (isTheFullFile==false)
758  fileListReceiver->partLength=partLength;
759 
760  FLR_MemoryBlock mb;
761  if (fileListReceiver->pushedFiles.Has(onFileStruct.fileIndex)==false)
762  {
763  if (onFileStruct.byteLengthOfThisFile <= SLNET_MAX_RETRIEVABLE_FILESIZE)
764  mb.flrMemoryBlock = (char*)rakMalloc_Ex(onFileStruct.byteLengthOfThisFile, _FILE_AND_LINE_);
765  else
766  mb.flrMemoryBlock = nullptr;

768  }
769  else
770  {
771  mb=fileListReceiver->pushedFiles.Get(onFileStruct.fileIndex);
772  }
773

775  unsigned int unreadBytes = BITS_TO_BYTES(unreadBits);

777  if (isTheFullFile)
778  amountToRead=chunkLength;
779  else
780  amountToRead=unreadBytes;
781 
782  inBitStream.AlignReadToByteBoundary();
783 
784  FileListTransferCBInterface::FileProgressStruct fps;
785 
786  if (isTheFullFile)
787  {
788  if (mb.flrMemoryBlock)
789  {
790  // Either the very first block, or a subsequent block and allocateIrIDataChunkAutomatically was true for the first block
791  memcpy(mb.flrMemoryBlock+offset, inBitStream.GetData()+BITS_TO_BYTES(inBitStream.GetReadOffset()), amountToRead);
792  fps.iriDataChunk=mb.flrMemoryBlock+offset;
793  }
794  else
795  {
796  // In here mb.flrMemoryBlock is null
797  // This means the first block explicitly deallocated the memory, and no blocks will be permanently held by RakNet
798  fps.iriDataChunk=(char*) inBitStream.GetData()+BITS_TO_BYTES(inBitStream.GetReadOffset());
799  }
800 
801  onFileStruct.bytesDownloadedForThisFile=offset+chunkLength;
802  fileListReceiver->setTotalDownloadedLength+=chunkLength;
803  onFileStruct.bytesDownloadedForThisSet=fileListReceiver->setTotalDownloadedLength;
804  }
805  else
806  {
807  onFileStruct.bytesDownloadedForThisFile=offset+partLength*partCount;
808  onFileStruct.bytesDownloadedForThisSet=fileListReceiver->setTotalDownloadedLength+partCount*partLength;
809  fps.iriDataChunk=(char*) inBitStream.GetData()+BITS_TO_BYTES(inBitStream.GetReadOffset());
810  }
811 
812  onFileStruct.numberOfFilesInThisSet=fileListReceiver->setCount;
813 // onFileStruct.setTotalCompressedTransmissionLength=fileListReceiver->setTotalCompressedTransmissionLength;
814  onFileStruct.byteLengthOfThisSet=fileListReceiver->setTotalFinalLength;
815  // Note: mb.flrMemoryBlock may be null here
816  onFileStruct.fileData=mb.flrMemoryBlock;
817  onFileStruct.senderSystemAddress=packet->systemAddress;
818  onFileStruct.senderGuid=packet->guid;
819 
820  unsigned int totalNotifications;
821  unsigned int currentNotificationIndex;
822  if (chunkLength==0 || chunkLength==onFileStruct.byteLengthOfThisFile)
823  totalNotifications=1;
824  else
825  totalNotifications = onFileStruct.byteLengthOfThisFile / chunkLength + 1;
826 
827  if (chunkLength==0)
828  currentNotificationIndex = 0;
829  else
830  currentNotificationIndex = offset / chunkLength;
831 
832  fps.onFileStruct=&onFileStruct;
833  fps.partCount=currentNotificationIndex;
834  fps.partTotal=totalNotifications;
835  fps.dataChunkLength=amountToRead;
836  fps.firstDataChunk=mb.flrMemoryBlock;
837  fps.allocateIrIDataChunkAutomatically=true;
838  fps.onFileStruct->fileData=mb.flrMemoryBlock;
839  fps.iriWriteOffset=offset;
840  fps.senderSystemAddress=packet->systemAddress;
841  fps.senderGuid=packet->guid;
842 
843  if (finished)
844  {
845  char *oldFileData=fps.onFileStruct->fileData;
846  if (fps.partCount==0)
847  fps.firstDataChunk=fps.iriDataChunk;
848  if (fps.partTotal==1)
849  fps.onFileStruct->fileData=fps.iriDataChunk;
850  fileListReceiver->downloadHandler->OnFileProgress(&fps);
851 
852  // Incremental read interface sent us a file chunk
853  // This is the last file chunk we were waiting for to consider the file done
854  if (fileListReceiver->downloadHandler->OnFile(&onFileStruct))
855  rakFree_Ex(oldFileData, _FILE_AND_LINE_ );
856  fileListReceiver->pushedFiles.Delete(onFileStruct.fileIndex);
857 
858  fileListReceiver->filesReceived++;
859 
860  // If this set is done, free the memory for it.
861  if ((int) fileListReceiver->setCount==fileListReceiver->filesReceived)
862  {
863  FileListTransferCBInterface::DownloadCompleteStruct dcs;
864  dcs.setID=fileListReceiver->setID;
865  dcs.numberOfFilesInThisSet=fileListReceiver->setCount;
866  dcs.byteLengthOfThisSet=fileListReceiver->setTotalFinalLength;
867  dcs.senderSystemAddress=packet->systemAddress;
868  dcs.senderGuid=packet->guid;
869 
870  if (fileListReceiver->downloadHandler->OnDownloadComplete(&dcs)==false)
871  {
872  fileListReceiver->downloadHandler->OnDereference();
873  fileListReceivers.Delete(onFileStruct.setID);
874  if (fileListReceiver->deleteDownloadHandler)
875  SLNet::OP_DELETE(fileListReceiver->downloadHandler, _FILE_AND_LINE_);
876  SLNet::OP_DELETE(fileListReceiver, _FILE_AND_LINE_);
877  }
878  }
879  }
880  else
881  {
882  if (isTheFullFile)
883  {
884  // 12/23/09 Don't use OnReferencePush anymore, just use OnFileProgress
885  fileListReceiver->downloadHandler->OnFileProgress(&fps);
886 
887  if (fps.allocateIrIDataChunkAutomatically==false)
888  {
889  rakFree_Ex(fileListReceiver->pushedFiles.Get(onFileStruct.fileIndex).flrMemoryBlock, _FILE_AND_LINE_ );
890  fileListReceiver->pushedFiles.Get(onFileStruct.fileIndex).flrMemoryBlock=0;
891  }
892  }
893  else
894  {
895  // This is a download progress notification for a file chunk using incremental read interface
896  // We don't have all the data for this chunk yet
897 
898  totalNotifications = onFileStruct.byteLengthOfThisFile / fileListReceiver->partLength + 1;
899  if (isTheFullFile==false)
900  currentNotificationIndex = (offset+partCount*fileListReceiver->partLength) / fileListReceiver->partLength ;
901  else
902  currentNotificationIndex = (offset+chunkLength) / fileListReceiver->partLength ;
903  unreadBytes = onFileStruct.byteLengthOfThisFile - ((currentNotificationIndex+1) * fileListReceiver->partLength);
904  fps.partCount=currentNotificationIndex;
905  fps.partTotal=totalNotifications;
906 
907 // 2/19/2013 Why was this check here? It prevent smaller progress notifications
908 // if (rakPeerInterface)
909  {
910  // Thus chunk is incomplete
911  fps.iriDataChunk=0;
912 
913  fileListReceiver->downloadHandler->OnFileProgress(&fps);
914  }
915  }
916  }
917 
918  return;

940 int SendIRIToAddressCB(FileListTransfer::ThreadData threadData, bool *returnOutput, void* perThreadData)
941 {
942  (void) perThreadData;
943 
944  FileListTransfer *fileListTransfer = threadData.fileListTransfer;
945  SystemAddress systemAddress = threadData.systemAddress;
946  unsigned short setId = threadData.setId;
947  *returnOutput=false;
948 
949  // Was previously using GetStatistics to get outgoing buffer size, but TCP with UnifiedSend doesn't have this
950  unsigned int bytesRead;
951  const char *dataBlocks[2];
952  int lengths[2];
953  unsigned int smallFileTotalSize=0;
954  SLNet::BitStream outBitstream;
955  unsigned int ftpIndex;
956 
957  fileListTransfer->fileToPushRecipientListMutex.Lock();
958  for (ftpIndex=0; ftpIndex < fileListTransfer->fileToPushRecipientList.Size(); ftpIndex++)
959  {
960  FileListTransfer::FileToPushRecipient *ftpr = fileListTransfer->fileToPushRecipientList[ftpIndex];
961  // Referenced by both ftpr and list
962  ftpr->AddRef();
963 
964  fileListTransfer->fileToPushRecipientListMutex.Unlock();
965 
966  if (ftpr->systemAddress==systemAddress && ftpr->setId==setId)
967  {
968  FileListTransfer::FileToPush *ftp;
970  ftp = ftpr->filesToPush.Pop();
972 
973  // Read and send chunk. If done, delete at this index
974  void *buff = rakMalloc_Ex(ftp->chunkSize, _FILE_AND_LINE_);
975  if (buff==0)
976  {
978  ftpr->filesToPush.PushAtHead(ftp,0,_FILE_AND_LINE_);
980 
981  ftpr->Deref();
982  notifyOutOfMemory(_FILE_AND_LINE_);
983  return 0;
984  }
985 
986  // Read the next file chunk
987  bytesRead=ftp->incrementalReadInterface->GetFilePart(ftp->fileListNode.fullPathToFile, ftp->currentOffset, ftp->chunkSize, buff, ftp->fileListNode.context);
988 
989  bool done = ftp->fileListNode.dataLengthBytes == ftp->currentOffset+bytesRead;
990  while (done && ftp->currentOffset==0 && smallFileTotalSize<ftp->chunkSize)
991  {
993  // The reason for 2 is that ID_FILE_LIST_REFERENCE_PUSH gets ID_FILE_LIST_REFERENCE_PUSH_ACK. WIthout ID_FILE_LIST_REFERENCE_PUSH_ACK, SendIRIToAddressCB would not be called again
994  if (ftpr->filesToPush.Size()<2)
995  {
997  break;
998  }
1000 
1001  // Send all small files at once, rather than wait for ID_FILE_LIST_REFERENCE_PUSH. But at least one ID_FILE_LIST_REFERENCE_PUSH must be sent
1002  outBitstream.Reset();
1003  outBitstream.Write((MessageID)ID_FILE_LIST_TRANSFER_FILE);
1004  // outBitstream.Write(ftp->fileListNode.context);
1005  outBitstream << ftp->fileListNode.context;
1006  outBitstream.Write(setId);
1007  StringCompressor::Instance()->EncodeString(ftp->fileListNode.filename, 512, &outBitstream);
1008  outBitstream.WriteCompressed(ftp->setIndex);
1009  outBitstream.WriteCompressed(ftp->fileListNode.dataLengthBytes); // Original length in bytes
1010  outBitstream.AlignWriteToByteBoundary();
1011  dataBlocks[0]=(char*) outBitstream.GetData();
1012  lengths[0]=outBitstream.GetNumberOfBytesUsed();
1013  dataBlocks[1]=(const char*) buff;
1014  lengths[1]=bytesRead;
1015 
1016  fileListTransfer->SendListUnified(dataBlocks,lengths,2,ftp->packetPriority, RELIABLE_ORDERED, ftp->orderingChannel, systemAddress, false);
1017 
1018  // LWS : fixed freed pointer reference
1019 // unsigned int chunkSize = ftp->chunkSize;
1020  SLNet::OP_DELETE(ftp,_FILE_AND_LINE_);
1021  smallFileTotalSize+=bytesRead;
1022  //done = bytesRead!=ftp->chunkSize;
1024  ftp = ftpr->filesToPush.Pop();
1026 
1027  bytesRead=ftp->incrementalReadInterface->GetFilePart(ftp->fileListNode.fullPathToFile, ftp->currentOffset, ftp->chunkSize, buff, ftp->fileListNode.context);
1028  done = ftp->fileListNode.dataLengthBytes == ftp->currentOffset+bytesRead;
1029  }
1030 
1031 
1032  outBitstream.Reset();
1033  outBitstream.Write((MessageID)ID_FILE_LIST_REFERENCE_PUSH);
1034  // outBitstream.Write(ftp->fileListNode.context);
1035  outBitstream << ftp->fileListNode.context;
1036  outBitstream.Write(setId);
1037  StringCompressor::Instance()->EncodeString(ftp->fileListNode.filename, 512, &outBitstream);
1038  outBitstream.WriteCompressed(ftp->setIndex);
1039  outBitstream.WriteCompressed(ftp->fileListNode.dataLengthBytes); // Original length in bytes
1040  outBitstream.WriteCompressed(ftp->currentOffset);
1041  ftp->currentOffset+=bytesRead;
1042  outBitstream.WriteCompressed(bytesRead);
1043  outBitstream.Write(done);
1044 
1045  for (unsigned int flpcIndex=0; flpcIndex < fileListTransfer->fileListProgressCallbacks.Size(); flpcIndex++)
1046  fileListTransfer->fileListProgressCallbacks[flpcIndex]->OnFilePush(ftp->fileListNode.filename, ftp->fileListNode.fileLengthBytes, ftp->currentOffset-bytesRead, bytesRead, done, systemAddress, setId);
1047 
1048  dataBlocks[0]=(char*) outBitstream.GetData();
1049  lengths[0]=outBitstream.GetNumberOfBytesUsed();
1050  dataBlocks[1]=(char*) buff;
1051  lengths[1]=bytesRead;
1052  //rakPeerInterface->SendList(dataBlocks,lengths,2,ftp->packetPriority, RELIABLE_ORDERED, ftp->orderingChannel, ftp->systemAddress, false);
1053  char orderingChannel = ftp->orderingChannel;
1054  PacketPriority packetPriority = ftp->packetPriority;
1055 
1056  // Mutex state: FileToPushRecipient (ftpr) has AddRef. fileToPushRecipientListMutex not locked.
1057  if (done)
1058  {
1059  // Done
1060  //unsigned short setId = ftp->setID;
1061  SLNet::OP_DELETE(ftp,_FILE_AND_LINE_);
1062 
1064  if (ftpr->filesToPush.Size()==0)
1065  {
1067

1070 
1071  // Remove ftpr from fileToPushRecipientList
1072  fileListTransfer->RemoveFromList(ftpr);
1073  }
1074  else
1075  {
1077  }
1078  }
1079  else
1080  {
1082  ftpr->filesToPush.PushAtHead(ftp,0,_FILE_AND_LINE_);
1084  }

1086  ftpr->Deref();

1090  fileListTransfer->SendListUnified(dataBlocks,lengths,2, packetPriority, RELIABLE_ORDERED, orderingChannel, systemAddress, false);
1091 
1092  rakFree_Ex(buff, _FILE_AND_LINE_ );
1093  return 0;
1094  }
1095  else
