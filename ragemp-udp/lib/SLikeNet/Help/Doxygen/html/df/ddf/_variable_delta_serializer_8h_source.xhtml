164  template <class VarType>
165  void SerializeVariable(SerializationContext *context, const VarType &variable)
166  {
167  if (context->newSystemSend)
168  {
169  if (context->variableHistory->variableListDeltaTracker.IsPastEndOfList()==false)
170  {
171  // previously sent data to another system
172  context->bitStream->Write(true);
173  context->bitStream->Write(variable);
174  context->anyVariablesWritten=true;
175  }
176  else
177  {
178  // never sent data to another system
179  context->variableHistory->variableListDeltaTracker.WriteVarToBitstream(variable, context->bitStream);
180  context->anyVariablesWritten=true;
181  }
182  }
183  else if (context->serializationMode==UNRELIABLE_WITH_ACK_RECEIPT)
184  {
185  context->anyVariablesWritten|=
186  context->variableHistory->variableListDeltaTracker.WriteVarToBitstream(variable, context->bitStream, context->changedVariables->bitField, context->changedVariables->bitWriteIndex++);
187  }
188  else
189  {
190  if (context->variableHistoryIdentical)
191  {
192  // Identical serialization to a number of systems
193  if (didComparisonThisTick==false)
194  context->anyVariablesWritten|=
195  context->variableHistory->variableListDeltaTracker.WriteVarToBitstream(variable, context->bitStream);
196  // Else bitstream is written to at the end
197  }
198  else
199  {
200  // Per-system serialization
201  context->anyVariablesWritten|=
202  context->variableHistory->variableListDeltaTracker.WriteVarToBitstream(variable, context->bitStream);
203  }
204  }
205  }
