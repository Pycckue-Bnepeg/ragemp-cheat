185 void RNS2_NativeClient::ProcessBufferedSend(void)
186 {
187  // Don't send until bound
188  if (bindState!=BS_BOUND)
189  return;
190  // Fast non-threadsafe check
191  if (bufferedSends.IsEmpty()==true)
192  return;
193 
194  sendInProgressMutex.Lock();
195  if (sendInProgress==true) {sendInProgressMutex.Unlock(); return;}
196  else {sendInProgress=true;}
197  sendInProgressMutex.Unlock();
198 
199  RNS2_SendParameters_NativeClient *sp;
200  bufferedSendsMutex.Lock();
201  if (bufferedSends.IsEmpty()==false)
202  sp=bufferedSends.Pop();
203  else
204  sp=0;
205  bufferedSendsMutex.Unlock();
206  if (sp==0)
207  {
208  sendInProgressMutex.Lock();
209  sendInProgress=false;
210  sendInProgressMutex.Unlock();
211  return; // Nothing to send after all
212  }
213 
214  SendImmediate(sp);
215  // sp remains in memory until the callback completes
216  // DeallocSP(sp);
217 }

351 unsigned RNS2_Berkley::RecvFromLoopInt(void)
352 {
353  isRecvFromLoopThreadActive.Increment();
354 
355  while ( endThreads == false )
356  {
357  RNS2RecvStruct *recvFromStruct;
358  recvFromStruct=binding.eventHandler->AllocRNS2RecvStruct(_FILE_AND_LINE_);
359  if (recvFromStruct != NULL)
360  {
361  recvFromStruct->socket=this;
362  RecvFromBlocking(recvFromStruct);
363 
364  if (recvFromStruct->bytesRead>0)
365  {
366  RakAssert(recvFromStruct->systemAddress.GetPort());
367  binding.eventHandler->OnRNS2Recv(recvFromStruct);
368  }
369  else
370  {
371  RakSleep(0);
372  binding.eventHandler->DeallocRNS2RecvStruct(recvFromStruct, _FILE_AND_LINE_);
373  }
374  }
375  }
376  isRecvFromLoopThreadActive.Decrement();
377 
378 
379 
380 
381  return 0;
382
