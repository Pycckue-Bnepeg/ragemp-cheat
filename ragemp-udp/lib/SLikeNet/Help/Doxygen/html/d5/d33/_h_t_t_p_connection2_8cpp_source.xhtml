153 void ReadChunkBlock( size_t &currentChunkSize, size_t &bytesReadSoFar, char *txtIn, RakString &txtOut)
154 {
155  size_t bytesToRead;
156  size_t sLen;
157 
158  do
159  {
160  bytesToRead = currentChunkSize - bytesReadSoFar;
161  sLen = strlen(txtIn);
162  if (sLen < bytesToRead)
163  bytesToRead = sLen;
164  txtOut.AppendBytes(txtIn, bytesToRead);
165  txtIn += bytesToRead;
166  bytesReadSoFar += bytesToRead;
167  if (*txtIn == 0)
168  {
169  // currentChunkSize=0;
170  return;
171  }
172  // char *newLine = strstr(txtIn, "\r\n");
173  if (txtIn[0] && txtIn[0]=='\r' && txtIn[1] && txtIn[1]=='\n' )
174  txtIn += 2; // Newline
175  char *newLine;
176  currentChunkSize = ReadChunkSize(txtIn, &newLine);
177  RakAssert(currentChunkSize < 50000); // Sanity check
178  if (currentChunkSize == 0)
179  return;
180  if (newLine == 0)
181  return;
182  bytesReadSoFar=0;
183  txtIn = newLine + 2;
184  } while (txtIn);

421 void HTTPConnection2::SendPendingRequestToConnectedSystem(SystemAddress sa)
422 {
423  if (sa==UNASSIGNED_SYSTEM_ADDRESS)
424  return;
425 
426  unsigned int requestsSent=0;
427 
428  // Search through requests to find a match for this instance of TCPInterface and SystemAddress
429  unsigned int i;
430  i=0;
431  pendingRequestsMutex.Lock();
432  while (i < pendingRequests.Size())
433  {
434  Request *request = pendingRequests[i];
435  if (request->hostEstimatedAddress==sa)
436  {
437  pendingRequests.RemoveAtIndex(i);
438  // Send this request
439  request->hostCompletedAddress=sa;
440 
441  sentRequestsMutex.Lock();
442  sentRequests.Push(request, _FILE_AND_LINE_);
443  sentRequestsMutex.Unlock();
444 
445  pendingRequestsMutex.Unlock();
446 
447 #if OPEN_SSL_CLIENT_SUPPORT==1

449  tcpInterface->StartSSLClient(sa);
450 #endif

456  }
457  else
458  {
459  i++;
460  }
461  }
462  pendingRequestsMutex.Unlock();
463 
464  if (requestsSent==0)
465  {
466  pendingRequestsMutex.Lock();
467  if (pendingRequests.Size() > 0)
468  {
469  // Just assign

471  pendingRequests.RemoveAtIndex(0);
472 
473  request->hostCompletedAddress=sa;
474 
475  sentRequestsMutex.Lock();
476  sentRequests.Push(request, _FILE_AND_LINE_);
