240 PluginReceiveResult FullyConnectedMesh2::OnReceive(Packet *packet)
241 {
242  switch (packet->data[0])
243  {
244  case ID_REMOTE_NEW_INCOMING_CONNECTION:
245  {
246  if (connectOnNewRemoteConnections)
247  ConnectToRemoteNewIncomingConnections(packet);
248  }
249  break;
250  case ID_FCM2_REQUEST_FCMGUID:
251  OnRequestFCMGuid(packet);
252  return RR_STOP_PROCESSING_AND_DEALLOCATE;
253  //case ID_FCM2_UPDATE_USER_CONTEXT:
254  // OnUpdateUserContext(packet);
255  // return RR_STOP_PROCESSING_AND_DEALLOCATE;
256  case ID_FCM2_RESPOND_CONNECTION_COUNT:
257  OnRespondConnectionCount(packet);
258  return RR_STOP_PROCESSING_AND_DEALLOCATE;
259  case ID_FCM2_INFORM_FCMGUID:
260  OnInformFCMGuid(packet);
261  return RR_STOP_PROCESSING_AND_DEALLOCATE;
262  case ID_FCM2_UPDATE_MIN_TOTAL_CONNECTION_COUNT:
263  OnUpdateMinTotalConnectionCount(packet);
264  return RR_STOP_PROCESSING_AND_DEALLOCATE;
265  case ID_FCM2_NEW_HOST:
266  if (packet->wasGeneratedLocally==false)
267  return RR_STOP_PROCESSING_AND_DEALLOCATE;
268  break;
269  case ID_FCM2_VERIFIED_JOIN_START:
270  return OnVerifiedJoinStart(packet);
271  case ID_FCM2_VERIFIED_JOIN_CAPABLE:
272  return OnVerifiedJoinCapable(packet);
273  case ID_FCM2_VERIFIED_JOIN_FAILED:
274  OnVerifiedJoinFailed(packet->guid, true);
275  return RR_CONTINUE_PROCESSING;
276  case ID_FCM2_VERIFIED_JOIN_ACCEPTED:
277  if (packet->wasGeneratedLocally==false)
278  OnVerifiedJoinAccepted(packet);
279  return RR_CONTINUE_PROCESSING;
280  case ID_FCM2_VERIFIED_JOIN_REJECTED:
281  OnVerifiedJoinRejected(packet);
282  return RR_CONTINUE_PROCESSING;
283 
284  case ID_NAT_TARGET_UNRESPONSIVE:
285  case ID_NAT_TARGET_NOT_CONNECTED:
286  case ID_NAT_CONNECTION_TO_TARGET_LOST:
287  {
288  SLNet::RakNetGUID g;
289  SLNet::BitStream b(packet->data, packet->length, false);
290  b.IgnoreBits(8); // Ignore the ID_...
291  b.Read(g);
292  UpdateVerifiedJoinInProgressMember(g, UNASSIGNED_RAKNET_GUID, JIPS_FAILED);
293  return RR_CONTINUE_PROCESSING;
294  }
295  case ID_NAT_PUNCHTHROUGH_FAILED:
296  UpdateVerifiedJoinInProgressMember(packet->guid, UNASSIGNED_RAKNET_GUID, JIPS_FAILED);
297  return RR_CONTINUE_PROCESSING;
298  }
299 
300  return RR_CONTINUE_PROCESSING;

320 {
321  (void) lostConnectionReason;
322  (void) systemAddress;
323  (void) rakNetGUID;
324 
325  unsigned int idx;
326  idx=0;
327  while (idx < joinsInProgress.Size())
328  {
329  if (joinsInProgress[idx]->requester==rakNetGUID)
330  {
331  Packet *p = AllocatePacketUnified(sizeof(MessageID)+sizeof(unsigned char));
332  p->data[0]=ID_FCM2_VERIFIED_JOIN_FAILED;
333  p->systemAddress=systemAddress;
334  p->systemAddress.systemIndex=(SystemIndex)-1;
335  p->guid=rakNetGUID;
336  p->wasGeneratedLocally=true;
337  rakPeerInterface->PushBackPacket(p, true);
338 
339  for (unsigned int j=0; j < joinsInProgress[idx]->vjipMembers.Size(); j++)
340  {
341  if ( joinsInProgress[idx]->vjipMembers[j].userData != 0)
342  {
343  SLNet::OP_DELETE(joinsInProgress[idx]->vjipMembers[j].userData, _FILE_AND_LINE_);
344  }
345  }
346 
347  SLNet::OP_DELETE(joinsInProgress[idx], _FILE_AND_LINE_);
348  joinsInProgress.RemoveAtIndex(idx);
349  }
350  else
351  {
352  idx++;
353  }
354  }
355 
356  UpdateVerifiedJoinInProgressMember(rakNetGUID, UNASSIGNED_RAKNET_GUID, JIPS_FAILED);
357 
358  for (idx=0; idx < fcm2ParticipantList.Size(); idx++)
359  {
360  if (fcm2ParticipantList[idx]->rakNetGuid==rakNetGUID)
361  {
362  fcm2ParticipantList[idx]=fcm2ParticipantList[fcm2ParticipantList.Size()-1];
363 #ifdef DEBUG_FCM2
364  printf("Popping participant %s\n", fcm2ParticipantList[fcm2ParticipantList.Size()-1].rakNetGuid.ToString());
365 #endif
366 
367  fcm2ParticipantList.Pop();
368  if (rakNetGUID==hostRakNetGuid && ourFCMGuid!=0)
369  {
370  if (fcm2ParticipantList.Size()==0)
371  {
372  hostRakNetGuid=rakPeerInterface->GetMyGUID();
373  hostFCM2Guid=ourFCMGuid;
374  }
375  else
376  {
377  CalculateHost(&hostRakNetGuid, &hostFCM2Guid);
378  }
379  PushNewHost(hostRakNetGuid, rakNetGUID);
380  }

382  }

536 void FullyConnectedMesh2::OnRequestFCMGuid(Packet *packet)

541  bsIn.Read(hasRemoteFCMGuid);
542  SLNet::TimeUS senderElapsedRuntime=0;
543  unsigned int remoteTotalConnectionCount=0;
544  FCM2Guid theirFCMGuid=0;
545  if (hasRemoteFCMGuid)
546  {
547  bsIn.Read(remoteTotalConnectionCount);
548  bsIn.Read(theirFCMGuid);
549  }

553  }

555  BitStream remoteContext;

557  AddParticipantInternal(packet->guid,theirFCMGuid, &remoteContext);
558  */
559  AddParticipantInternal(packet->guid,theirFCMGuid);
560  if (ourFCMGuid==0)
561  {
562  if (hasRemoteFCMGuid==false)
563  {
564  // Nobody has a fcmGuid
565 
566  SLNet::TimeUS ourElapsedRuntime = GetElapsedRuntime();
567  if (ourElapsedRuntime>senderElapsedRuntime)
568  {
569  // We are probably host
570  SendConnectionCountResponse(packet->systemAddress, 2);
571  }
572  else
573  {

575  SendConnectionCountResponse(packet->systemAddress, 1);
576  }
577  }
578  else
579  {
580  // They have a fcmGuid, we do not
581  IncrementTotalConnectionCount(remoteTotalConnectionCount+1);
582 
583  AssignOurFCMGuid();
584  unsigned int idx;

586  SendOurFCMGuid(rakPeerInterface->GetSystemAddressFromGuid(fcm2ParticipantList[idx]->rakNetGuid));
587  }
588  }
589  else
590  {
591  if (hasRemoteFCMGuid==false)
592  {

594  SendConnectionCountResponse(packet->systemAddress, totalConnectionCount+1);
595  }
596  else
