134 UDPForwarderResult UDPForwarder::StartForwarding(SystemAddress source, SystemAddress destination, SLNet::TimeMS timeoutOnNoDataMS, const char *forceHostAddress, unsigned short socketFamily,
135  unsigned short *forwardingPort, __UDPSOCKET__ *forwardingSocket)
136 {
137  // Invalid parameters?
138  if (timeoutOnNoDataMS == 0 || timeoutOnNoDataMS > UDP_FORWARDER_MAXIMUM_TIMEOUT || source==UNASSIGNED_SYSTEM_ADDRESS || destination==UNASSIGNED_SYSTEM_ADDRESS)
139  return UDPFORWARDER_INVALID_PARAMETERS;
140 
141  if (isRunning.GetValue()==0)
142  return UDPFORWARDER_NOT_RUNNING;
143 
144  (void) socketFamily;
145 
146  unsigned int inputId = nextInputId++;
147 
148  StartForwardingInputStruct *sfis;
149  sfis = startForwardingInput.Allocate(_FILE_AND_LINE_);
150  sfis->source=source;
151  sfis->destination=destination;
152  sfis->timeoutOnNoDataMS=timeoutOnNoDataMS;
153  RakAssert(timeoutOnNoDataMS!=0);
154  if (forceHostAddress && forceHostAddress[0])
155  sfis->forceHostAddress=forceHostAddress;
156  sfis->socketFamily=socketFamily;
157  sfis->inputId=inputId;
158  startForwardingInput.Push(sfis);
159 
160  for(;;)
161  {
162  RakSleep(0);
163  startForwardingOutputMutex.Lock();
164  for (unsigned int i=0; i < startForwardingOutput.Size(); i++)
165  {
166  if (startForwardingOutput[i].inputId==inputId)
167  {
168  if (startForwardingOutput[i].result==UDPFORWARDER_SUCCESS)
169  {
170  if (forwardingPort)
171  *forwardingPort = startForwardingOutput[i].forwardingPort;
172  if (forwardingSocket)
173  *forwardingSocket = startForwardingOutput[i].forwardingSocket;
174  }
175  UDPForwarderResult res = startForwardingOutput[i].result;
176  startForwardingOutput.RemoveAtIndex(i);
177  startForwardingOutputMutex.Unlock();
178  return res;
179  }
180  }
181  startForwardingOutputMutex.Unlock();
182  }

401 void UDPForwarder::UpdateUDPForwarder(void)
402 {
403  /*
404 #if !defined(SN_TARGET_PSP2)
405  timeval tv;
406  tv.tv_sec=0;
407  tv.tv_usec=0;
408 #endif
409  */
410 
411  SLNet::TimeMS curTime = SLNet::GetTimeMS();
412 
413  StartForwardingInputStruct *sfis;
414  StartForwardingOutputStruct sfos;
415  sfos.forwardingSocket=INVALID_SOCKET;
416  sfos.forwardingPort=0;
417  sfos.inputId=0;
418  sfos.result=UDPFORWARDER_RESULT_COUNT;
419 
420  for(;;)
421  {
422  sfis = startForwardingInput.Pop();
423  if (sfis==0)
424  break;
425 
426  if (GetUsedForwardEntries()>maxForwardEntries)
427  {
428  sfos.result=UDPFORWARDER_NO_SOCKETS;
429  }
430  else
431  {
432  sfos.result=UDPFORWARDER_RESULT_COUNT;
433 
434  for (unsigned int i=0; i < forwardListNotUpdated.Size(); i++)
435  {
436  if (
437  (forwardListNotUpdated[i]->addr1Unconfirmed==sfis->source &&
438  forwardListNotUpdated[i]->addr2Unconfirmed==sfis->destination)
439  ||
440  (forwardListNotUpdated[i]->addr1Unconfirmed==sfis->destination &&
441  forwardListNotUpdated[i]->addr2Unconfirmed==sfis->source)
442  )
443  {
444  ForwardEntry *fe = forwardListNotUpdated[i];
445  sfos.forwardingPort = SocketLayer::GetLocalPort ( fe->socket );
446  sfos.forwardingSocket=fe->socket;
447  sfos.result=UDPFORWARDER_FORWARDING_ALREADY_EXISTS;
448  break;
449  }
450  }
451 
452  if (sfos.result==UDPFORWARDER_RESULT_COUNT)
453  {
454  int sock_opt;
455  sockaddr_in listenerSocketAddress;
456  listenerSocketAddress.sin_port = 0;
457  ForwardEntry *fe = SLNet::OP_NEW<UDPForwarder::ForwardEntry>(_FILE_AND_LINE_);
458  fe->addr1Unconfirmed=sfis->source;
459  fe->addr2Unconfirmed=sfis->destination;
460  fe->timeoutOnNoDataMS=sfis->timeoutOnNoDataMS;
461 
462 #if RAKNET_SUPPORT_IPV6!=1
463  fe->socket = socket__( AF_INET, SOCK_DGRAM, 0 );
464  listenerSocketAddress.sin_family = AF_INET;
465  if (sfis->forceHostAddress.IsEmpty()==false)
466  {
467 
468 
469 
470 
471  inet_pton(AF_INET, sfis->forceHostAddress.C_String(), &listenerSocketAddress.sin_addr.s_addr);
472 
473  }
474  else
475  {
476  listenerSocketAddress.sin_addr.s_addr = INADDR_ANY;
477  }
478  int ret = bind__( fe->socket, ( struct sockaddr * ) & listenerSocketAddress, sizeof( listenerSocketAddress ) );
479  if (ret==-1)
480  {
481  SLNet::OP_DELETE(fe,_FILE_AND_LINE_);
482  sfos.result=UDPFORWARDER_BIND_FAILED;
483  }
484  else
485  {
486  sfos.result=UDPFORWARDER_SUCCESS;
487  }
488 
489 #else // RAKNET_SUPPORT_IPV6==1
490  struct addrinfo hints;
491  memset(&hints, 0, sizeof (addrinfo)); // make sure the struct is empty
492  hints.ai_family = sfis->socketFamily;
493  hints.ai_socktype = SOCK_DGRAM; // UDP sockets
494  hints.ai_flags = AI_PASSIVE; // fill in my IP for me
495  struct addrinfo *servinfo=0, *aip; // will point to the results
496 
497  if (sfis->forceHostAddress.IsEmpty() || sfis->forceHostAddress=="UNASSIGNED_SYSTEM_ADDRESS")
498  getaddrinfo(0, "0", &hints, &servinfo);
499  else
500  getaddrinfo(sfis->forceHostAddress.C_String(), "0", &hints, &servinfo);
501 
502  for (aip = servinfo; aip != NULL; aip = aip->ai_next)
503  {
504  // Open socket. The address type depends on what
505  // getaddrinfo() gave us.
506  fe->socket = socket__(aip->ai_family, aip->ai_socktype, aip->ai_protocol);
507  if (fe->socket != INVALID_SOCKET)
508  {
509  int ret = bind__( fe->socket, aip->ai_addr, (int) aip->ai_addrlen );
510  if (ret>=0)
511  {
512  break;
513  }
514  else
515  {
516  closesocket__(fe->socket);
517  fe->socket=INVALID_SOCKET;
518  }
519  }
520  }
521 
522  if (fe->socket==INVALID_SOCKET)
523  sfos.result=UDPFORWARDER_BIND_FAILED;
524  else
525  sfos.result=UDPFORWARDER_SUCCESS;
526 #endif // RAKNET_SUPPORT_IPV6==1
527 
528  if (sfos.result==UDPFORWARDER_SUCCESS)
529  {
530  sfos.forwardingPort = SocketLayer::GetLocalPort ( fe->socket );
531  sfos.forwardingSocket=fe->socket;
532 
533  sock_opt=1024*256;
534  setsockopt__(fe->socket, SOL_SOCKET, SO_RCVBUF, ( char * ) & sock_opt, sizeof ( sock_opt ) );
535  sock_opt=0;
536  setsockopt__(fe->socket, SOL_SOCKET, SO_LINGER, ( char * ) & sock_opt, sizeof ( sock_opt ) );
537 #ifdef _WIN32
538  unsigned long nonblocking = 1;
539  ioctlsocket__( fe->socket, FIONBIO, &nonblocking );
540 
541 
542 
543 #else
544  fcntl( fe->socket, F_SETFL, O_NONBLOCK );
545 #endif
546 
547  forwardListNotUpdated.Insert(fe,_FILE_AND_LINE_);
548  }
549  }
550  }
551 
552  // Push result
553  sfos.inputId=sfis->inputId;
554  startForwardingOutputMutex.Lock();
555  startForwardingOutput.Push(sfos,_FILE_AND_LINE_);
556  startForwardingOutputMutex.Unlock();
557 
558  startForwardingInput.Deallocate(sfis, _FILE_AND_LINE_);
559  }
560 
561  StopForwardingStruct *sfs;
562 
563  for(;;)
564  {
565  sfs = stopForwardingCommands.Pop();
566  if (sfs==0)
567  break;
568 
569  ForwardEntry *fe;
570  for (unsigned int i=0; i < forwardListNotUpdated.Size(); i++)
571  {
572  if (
573  (forwardListNotUpdated[i]->addr1Unconfirmed==sfs->source &&
574  forwardListNotUpdated[i]->addr2Unconfirmed==sfs->destination)
575  ||
576  (forwardListNotUpdated[i]->addr1Unconfirmed==sfs->destination &&
577  forwardListNotUpdated[i]->addr2Unconfirmed==sfs->source)
578  )
579  {
580  fe = forwardListNotUpdated[i];
581  forwardListNotUpdated.RemoveAtIndexFast(i);
582  SLNet::OP_DELETE(fe, _FILE_AND_LINE_);
583  break;
584  }
585  }
586 
587  stopForwardingCommands.Deallocate(sfs, _FILE_AND_LINE_);
588  }
589 
590  unsigned int i;
591 
592  i=0;
593  while (i < forwardListNotUpdated.Size())
594  {
595  if (curTime > forwardListNotUpdated[i]->timeLastDatagramForwarded && // Account for timestamp wrap
596  curTime > forwardListNotUpdated[i]->timeLastDatagramForwarded+forwardListNotUpdated[i]->timeoutOnNoDataMS)
597  {
598  SLNet::OP_DELETE(forwardListNotUpdated[i],_FILE_AND_LINE_);
599  forwardListNotUpdated.RemoveAtIndex(i);
600  }
601  else
602  i++;
603  }
604 
605  ForwardEntry *forwardEntry;
606  for (i=0; i < forwardListNotUpdated.Size(); i++)
607  {
608  forwardEntry = forwardListNotUpdated[i];
609  RecvFrom(curTime, forwardEntry);
610  }
611 }
