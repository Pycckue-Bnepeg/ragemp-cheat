287 void MessageFilter::Update(void)
288 {
289  // Update all timers for all systems. If those systems' filter sets are expired, take the appropriate action.
290  SLNet::Time curTime = SLNet::GetTime();
291  if (GreaterThan(curTime - 1000, whenLastTimeoutCheck))
292  {
293  DataStructures::List< FilteredSystem > itemList;
294  DataStructures::List< AddressOrGUID > keyList;
295  systemList.GetAsList(itemList, keyList, _FILE_AND_LINE_);
296 
297  unsigned int index;
298  for (index=0; index < itemList.Size(); index++)
299  {
300  if (itemList[index].filter &&
301  itemList[index].filter->maxMemberTimeMS>0 &&
302  curTime-itemList[index].timeEnteredThisSet >= itemList[index].filter->maxMemberTimeMS)
303  {
304  if (itemList[index].filter->timeoutCallback)
305  itemList[index].filter->timeoutCallback(rakPeerInterface, keyList[index], itemList[index].filter->filterSetID, itemList[index].filter->timeoutUserData);
306 
307  if (itemList[index].filter->banOnFilterTimeExceed && rakPeerInterface)
308  {
309  char str1[64];
310  keyList[index].ToString(false, str1, 64);
311  rakPeerInterface->AddToBanList(str1, itemList[index].filter->timeExceedBanTimeMS);
312  }
313  if (rakPeerInterface)
314  rakPeerInterface->CloseConnection(keyList[index], true, 0);
315 #if _RAKNET_SUPPORT_PacketizedTCP==1 && _RAKNET_SUPPORT_TCPInterface==1
316  else
317  tcpInterface->CloseConnection(keyList[index].systemAddress);
318 #endif
319 
320  systemList.Remove(keyList[index], _FILE_AND_LINE_);
321  }
322  }
323 
324  whenLastTimeoutCheck=curTime+1000;
325  }

353  PluginReceiveResult MessageFilter::OnReceive(Packet *packet)
354 {
355  DataStructures::HashIndex index;
356  unsigned char messageId;
357 
358  switch (packet->data[0])
359  {
360  case ID_NEW_INCOMING_CONNECTION:
361  case ID_CONNECTION_REQUEST_ACCEPTED:
362  case ID_CONNECTION_LOST:
363  case ID_DISCONNECTION_NOTIFICATION:
364  case ID_CONNECTION_ATTEMPT_FAILED:
365  case ID_NO_FREE_INCOMING_CONNECTIONS:
366  case ID_IP_RECENTLY_CONNECTED:
367  case ID_CONNECTION_BANNED:
368  case ID_INVALID_PASSWORD:
369  case ID_UNCONNECTED_PONG:
370  case ID_ALREADY_CONNECTED:
371  case ID_ADVERTISE_SYSTEM:
372  case ID_REMOTE_DISCONNECTION_NOTIFICATION:
373  case ID_REMOTE_CONNECTION_LOST:
374  case ID_REMOTE_NEW_INCOMING_CONNECTION:
375  case ID_DOWNLOAD_PROGRESS:
376  break;
377  default:
378  if (packet->data[0]==ID_TIMESTAMP)
379  {
380  if (packet->length<sizeof(MessageID) + sizeof(SLNet::TimeMS))
381  return RR_STOP_PROCESSING_AND_DEALLOCATE; // Invalid message
382  messageId=packet->data[sizeof(MessageID) + sizeof(SLNet::TimeMS)];
383  }
384  else
385  messageId=packet->data[0];
386  // If this system is filtered, check if this message is allowed. If not allowed, return RR_STOP_PROCESSING_AND_DEALLOCATE
387  // index = systemList.GetIndexFromKey(packet->addressOrGUID, &objectExists);
388  index = systemList.GetIndexOf(packet);
389  if (index.IsInvalid())
390  break;
391  if (systemList.ItemAtIndex(index).filter->allowedIDs[messageId]==false)
392  {
393  OnInvalidMessage(systemList.ItemAtIndex(index).filter, packet, packet->data[0]);
394  return RR_STOP_PROCESSING_AND_DEALLOCATE;
395  }
396  if (packet->data[0]==ID_RPC_PLUGIN)
397  {
398  SLNet::BitStream bsIn(packet->data,packet->length,false);
399  bsIn.IgnoreBytes(2);
400  SLNet::RakString functionName;
401  bsIn.ReadCompressed(functionName);
402  if (systemList.ItemAtIndex(index).filter->allowedRPC4.HasData(functionName)==false)
403  {
404  OnInvalidMessage(systemList.ItemAtIndex(index).filter, packet, packet->data[0]);
405  return RR_STOP_PROCESSING_AND_DEALLOCATE;
406  }
407  }
408 
409  break;
410  }
411 
412  return RR_CONTINUE_PROCESSING;
413 }
