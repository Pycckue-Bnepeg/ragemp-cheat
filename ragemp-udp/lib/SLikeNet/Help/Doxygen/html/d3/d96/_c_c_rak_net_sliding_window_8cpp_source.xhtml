133 bool CCRakNetSlidingWindow::OnGotPacket(DatagramSequenceNumberType datagramSequenceNumber, bool isContinuousSend, CCTimeType curTime, uint32_t sizeInBytes, uint32_t *skippedMessageCount)
134 {
135  (void) curTime;
136  (void) sizeInBytes;
137  (void) isContinuousSend;
138 
139  if (oldestUnsentAck==0)
140  oldestUnsentAck=curTime;
141 
142  if (datagramSequenceNumber==expectedNextSequenceNumber)
143  {
144  *skippedMessageCount=0;
145  expectedNextSequenceNumber=datagramSequenceNumber+(DatagramSequenceNumberType)1;
146  }
147  else if (GreaterThan(datagramSequenceNumber, expectedNextSequenceNumber))
148  {
149  *skippedMessageCount=datagramSequenceNumber-expectedNextSequenceNumber;
150  // Sanity check, just use timeout resend if this was really valid
151  if (*skippedMessageCount>1000)
152  {
153  // During testing, the nat punchthrough server got 51200 on the first packet. I have no idea where this comes from, but has happened twice
154  if (*skippedMessageCount>(uint32_t)50000)
155  return false;
156  *skippedMessageCount=1000;
157  }
158  expectedNextSequenceNumber=datagramSequenceNumber+(DatagramSequenceNumberType)1;
159  }
160  else
161  {
162  *skippedMessageCount=0;
163  }
164 
165  return true;
166 }

168 void CCRakNetSlidingWindow::OnResend(CCTimeType curTime, SLNet::TimeUS nextActionTime)
169 {
170  (void) curTime;
171  (void) nextActionTime;
172 
173  if (_isContinuousSend && backoffThisBlock==false && cwnd>MAXIMUM_MTU_INCLUDING_UDP_HEADER*2)
174  {
175  // Spec says 1/2 cwnd, but it never recovers because cwnd increases too slowly
176  //ssThresh=cwnd-8.0 * (MAXIMUM_MTU_INCLUDING_UDP_HEADER*MAXIMUM_MTU_INCLUDING_UDP_HEADER/cwnd);
177  ssThresh=cwnd/2;
178  if (ssThresh<MAXIMUM_MTU_INCLUDING_UDP_HEADER)
179  ssThresh=MAXIMUM_MTU_INCLUDING_UDP_HEADER;
180  cwnd=MAXIMUM_MTU_INCLUDING_UDP_HEADER;
181 
182  // Only backoff once per period
183  nextCongestionControlBlock=nextDatagramSequenceNumber;
184  backoffThisBlock=true;
185 
186  // CC PRINTF
187  //printf("-- %.0f (Resend) Enter slow start.\n", cwnd);
188  }
189 }

206 void CCRakNetSlidingWindow::OnAck(CCTimeType curTime, CCTimeType rtt, bool hasBAndAS, BytesPerMicrosecond _B, BytesPerMicrosecond _AS, double totalUserDataBytesAcked, bool isContinuousSend, DatagramSequenceNumberType sequenceNumber )
207 {
208  (void) _B;
209  (void) totalUserDataBytesAcked;
210  (void) _AS;
211  (void) hasBAndAS;
212  (void) curTime;
213  (void) rtt;
214 
215  lastRtt=(double) rtt;
216  if (estimatedRTT==UNSET_TIME_US)
217  {
218  estimatedRTT=(double) rtt;
219  deviationRtt=(double)rtt;
220  }
221  else
222  {
223  double d = .05;
224  double difference = rtt - estimatedRTT;
225  estimatedRTT = estimatedRTT + d * difference;
226  deviationRtt = deviationRtt + d * (std::abs(difference) - deviationRtt);
227  }
228 
229  _isContinuousSend=isContinuousSend;
230 
231  if (isContinuousSend==false)
232  return;
233 
234  bool isNewCongestionControlPeriod;
235  isNewCongestionControlPeriod = GreaterThan(sequenceNumber, nextCongestionControlBlock);
236 
237  if (isNewCongestionControlPeriod)
238  {
239  backoffThisBlock=false;
240  speedUpThisBlock=false;
241  nextCongestionControlBlock=nextDatagramSequenceNumber;
242  }
243 
244  if (IsInSlowStart())
245  {
246  cwnd+=MAXIMUM_MTU_INCLUDING_UDP_HEADER;
247  if (cwnd > ssThresh && ssThresh!=0)
248  cwnd = ssThresh + MAXIMUM_MTU_INCLUDING_UDP_HEADER*MAXIMUM_MTU_INCLUDING_UDP_HEADER/cwnd;
249 
250  // CC PRINTF
251  // printf("++ %.0f Slow start increase.\n", cwnd);
252 
253  }
254  else if (isNewCongestionControlPeriod)
255  {
256  cwnd+=MAXIMUM_MTU_INCLUDING_UDP_HEADER*MAXIMUM_MTU_INCLUDING_UDP_HEADER/cwnd;
257 
258  // CC PRINTF
259  // printf("+ %.0f Congestion avoidance increase.\n", cwnd);
260  }
261 }

293 CCTimeType CCRakNetSlidingWindow::GetRTOForRetransmission(unsigned char timesSent) const
294 {
295  (void) timesSent;
296 
297 #if CC_TIME_TYPE_BYTES==4
298  const CCTimeType maxThreshold=2000;
299  //const CCTimeType minThreshold=100;
300  const CCTimeType additionalVariance=30;
301 #else
302  const CCTimeType maxThreshold=2000000;
303  //const CCTimeType minThreshold=100000;
304  const CCTimeType additionalVariance=30000;
305 #endif
306 
307 
308  if (estimatedRTT==UNSET_TIME_US)
309  return maxThreshold;
310 
311  //double u=1.0f;
312  double u=2.0f;
313  double q=4.0f;
314 
315  CCTimeType threshhold = (CCTimeType) (u * estimatedRTT + q * deviationRtt) + additionalVariance;
316  if (threshhold > maxThreshold)
317  return maxThreshold;
318  return threshhold;
319 }
