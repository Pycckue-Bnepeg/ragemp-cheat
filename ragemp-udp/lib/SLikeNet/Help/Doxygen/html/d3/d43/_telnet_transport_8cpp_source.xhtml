68 void TelnetTransport::Send( SystemAddress systemAddress, const char *data,... )
69 {
70  if (tcpInterface==0) return;
71 
72  if (data==0 || data[0]==0)
73  return;
74 
75  char text[REMOTE_MAX_TEXT_INPUT];
76  size_t prefixLength;
77  if (sendPrefix)
78  {
79  strcpy_s(text, sendPrefix);
80  prefixLength = strlen(sendPrefix);
81  }
82  else
83  {
84  text[0]=0;
85  prefixLength=0;
86  }
87  va_list ap;
88  va_start(ap, data);
89  vsnprintf_s(text+prefixLength, REMOTE_MAX_TEXT_INPUT-prefixLength, REMOTE_MAX_TEXT_INPUT-prefixLength-1, data, ap);
90  va_end(ap);
91 
92  if (sendSuffix)
93  {
94  size_t length = strlen(text);
95  size_t availableChars = REMOTE_MAX_TEXT_INPUT-length-1;
96  strncat_s(text, sendSuffix, availableChars);
97  }
98 
99  tcpInterface->Send(text, (unsigned int) strlen(text), systemAddress, false);
100 }

105 Packet* TelnetTransport::Receive( void )
106 {
107  if (tcpInterface==0) return 0;
108  Packet *p = tcpInterface->Receive();
109  if (p==0)
110  return 0;
111 
112  /*
113  if (p->data[0]==255)
114  {
115  unsigned i;
116  for (i=0; i < p->length; i++)
117  {
118  RAKNET_DEBUG_PRINTF("%i ", p->data[i]);

125 
126  // Get this guy's cursor buffer. This is real bullcrap that I have to do this.
127  unsigned i;
128  TelnetClient *remoteClient=0;
129  for (i=0; i < remoteClients.Size(); i++)
130  {
131  if (remoteClients[i]->systemAddress==p->systemAddress)
132  remoteClient=remoteClients[i];
133  }
134  //RakAssert(remoteClient);
135  if (remoteClient==0)
136  {
137  tcpInterface->DeallocatePacket(p);
138  return 0;
139  }
140 
141 
142  if (p->length==3 && p->data[0]==27 && p->data[1]==91 && p->data[2]==65)
143  {
144  if (remoteClient->lastSentTextInput[0])
145  {
146  // Up arrow, return last string
147  for (i=0; remoteClient->textInput[i]; i++)
148  remoteClient->textInput[i]=8;
149  strcat_s(remoteClient->textInput, remoteClient->lastSentTextInput);
150  tcpInterface->Send((const char *)remoteClient->textInput, (unsigned int) strlen(remoteClient->textInput), p->systemAddress, false);
151  strcpy_s(remoteClient->textInput,remoteClient->lastSentTextInput);
152  remoteClient->cursorPosition=(unsigned int) strlen(remoteClient->textInput);
153  }
154 
155  return 0;
156  }
157 
158 
159  // 127 is delete - ignore that
160  // 9 is tab
161  // 27 is escape
162  if (p->data[0]>=127 || p->data[0]==9 || p->data[0]==27)
163  {
164  tcpInterface->DeallocatePacket(p);
165  return 0;
166  }
167 
168  // Hack - I don't know what the hell this is about but cursor keys send 3 characters at a time. I can block these
169  //Up=27,91,65
170  //Down=27,91,66
171  //Right=27,91,67
172  //Left=27,91,68
173  if (p->length==3 && p->data[0]==27 && p->data[1]==91 && p->data[2]>=65 && p->data[2]<=68)
174  {
175  tcpInterface->DeallocatePacket(p);
176  return 0;
177  }
178 
179 
180 
181  // Echo
182 #ifdef ECHO_INPUT
183  tcpInterface->Send((const char *)p->data, p->length, p->systemAddress, false);
184 #endif
185 
186  bool gotLine;
187  // Process each character in turn
188  for (i=0; i < p->length; i++)
189  {
190 
191 #ifdef ECHO_INPUT
192  if (p->data[i]==8)
193  {
194  char spaceThenBack[2];
195  spaceThenBack[0]=' ';
196  spaceThenBack[1]=8;
197  tcpInterface->Send((const char *)spaceThenBack, 2, p->systemAddress, false);
198  }
199 #endif
200 
201  gotLine=ReassembleLine(remoteClient, p->data[i]);
202  if (gotLine && remoteClient->textInput[0])
203  {
204 
205  Packet *reassembledLine = (Packet*) rakMalloc_Ex(sizeof(Packet), _FILE_AND_LINE_);
206  reassembledLine->length=(unsigned int) strlen(remoteClient->textInput);

214  reassembledLine->data[reassembledLine->length]=0;
215  reassembledLine->systemAddress=p->systemAddress;

243  //command[1]=253; // WON'T

249  // Tell the other side to use line mode
250  // http://www.faqs.org/rfcs/rfc1184.html
251  // IAC DO LINEMODE
252  // command[0]=255; // IAC
253  // command[1]=252; // DO
254  // command[2]=34; // LINEMODE
255  // tcpInterface->Send((const char*)command, 3, newConnection);
256 
257  */
258 
259  TelnetClient *remoteClient=0;
260  for (i=0; i < remoteClients.Size(); i++)
261  {
262  if (remoteClients[i]->systemAddress==newConnection)
263  {
264  remoteClient=remoteClients[i];
265  remoteClient->cursorPosition=0;
266  }
267  }
268

338 bool TelnetTransport::ReassembleLine(TelnetTransport::TelnetClient* remoteClient, unsigned char c)
339 {
340  if (c=='\n')
341  {
342  remoteClient->textInput[remoteClient->cursorPosition]=0;
343  remoteClient->cursorPosition=0;
344 #ifdef _PRINTF_DEBUG
345  RAKNET_DEBUG_PRINTF("[Done] %s\n", remoteClient->textInput);
346 #endif
347  return true;
348  }
349  else if (c==8) // backspace
350  {
351  if (remoteClient->cursorPosition>0)
352  {
353  remoteClient->textInput[--remoteClient->cursorPosition]=0;
354 #ifdef _PRINTF_DEBUG
355  RAKNET_DEBUG_PRINTF("[Back] %s\n", remoteClient->textInput);
356 #endif
357  }
358  }
359  else if (c>=32 && c <127)
360  {
361  if (remoteClient->cursorPosition < REMOTE_MAX_TEXT_INPUT)
362  {
363  remoteClient->textInput[remoteClient->cursorPosition++]=c;
364 #ifdef _PRINTF_DEBUG
365  RAKNET_DEBUG_PRINTF("[Norm] %s\n", remoteClient->textInput);
366 #endif
367  }
368  }
369  return false;
370 }
