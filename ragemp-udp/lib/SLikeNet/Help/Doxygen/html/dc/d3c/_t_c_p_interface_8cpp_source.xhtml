351 SystemAddress TCPInterface::Connect(const char* host, unsigned short remotePort, bool block, unsigned short socketFamily, const char *bindAddress)
352 {
353  if (threadRunning.GetValue()==0)
354  return UNASSIGNED_SYSTEM_ADDRESS;
355 
356  int newRemoteClientIndex=-1;
357  for (newRemoteClientIndex=0; newRemoteClientIndex < remoteClientsLength; newRemoteClientIndex++)
358  {
359  remoteClients[newRemoteClientIndex].isActiveMutex.Lock();
360  if (remoteClients[newRemoteClientIndex].isActive==false)
361  {
362  remoteClients[newRemoteClientIndex].SetActive(true);
363  remoteClients[newRemoteClientIndex].isActiveMutex.Unlock();
364  break;
365  }
366  remoteClients[newRemoteClientIndex].isActiveMutex.Unlock();
367  }
368  if (newRemoteClientIndex==-1)
369  return UNASSIGNED_SYSTEM_ADDRESS;
370 
371  if (block)
372  {
373  SystemAddress systemAddress;
374  systemAddress.FromString(host);
375  systemAddress.SetPortHostOrder(remotePort);
376  systemAddress.systemIndex=(SystemIndex) newRemoteClientIndex;
377  char buffout[128];
378  systemAddress.ToString(false,buffout, static_cast<size_t>(128));
379 
380  __TCPSOCKET__ sockfd = SocketConnect(buffout, remotePort, socketFamily, bindAddress);
381  // Windows RT TODO
382 #if !defined(WINDOWS_STORE_RT)
383  if (sockfd==0)
384 #endif
385  {
386  remoteClients[newRemoteClientIndex].isActiveMutex.Lock();
387  remoteClients[newRemoteClientIndex].SetActive(false);
388  remoteClients[newRemoteClientIndex].isActiveMutex.Unlock();
389 
390  failedConnectionAttemptMutex.Lock();
391  failedConnectionAttempts.Push(systemAddress, _FILE_AND_LINE_ );
392  failedConnectionAttemptMutex.Unlock();
393 
394  return UNASSIGNED_SYSTEM_ADDRESS;
395  }
396 
397  remoteClients[newRemoteClientIndex].socket=sockfd;
398  remoteClients[newRemoteClientIndex].systemAddress=systemAddress;
399 
400  completedConnectionAttemptMutex.Lock();
401  completedConnectionAttempts.Push(remoteClients[newRemoteClientIndex].systemAddress, _FILE_AND_LINE_ );
402  completedConnectionAttemptMutex.Unlock();
403 
404  return remoteClients[newRemoteClientIndex].systemAddress;
405  }
406  else
407  {
408  ThisPtrPlusSysAddr *s = SLNet::OP_NEW<ThisPtrPlusSysAddr>( _FILE_AND_LINE_ );
409  s->systemAddress.FromStringExplicitPort(host,remotePort);
410  s->systemAddress.systemIndex=(SystemIndex) newRemoteClientIndex;
411  if (bindAddress)
412  strcpy_s(s->bindAddress, bindAddress);
413  else
414  s->bindAddress[0]=0;
415  s->tcpInterface=this;
416  s->socketFamily=socketFamily;
417 
418  // Start the connection thread
419  int errorCode;
420 
421 
422 
423 
424  errorCode = SLNet::RakThread::Create(ConnectionAttemptLoop, s, threadPriority);
425 
426  if (errorCode!=0)
427  {
428  SLNet::OP_DELETE(s, _FILE_AND_LINE_);
429  failedConnectionAttempts.Push(s->systemAddress, _FILE_AND_LINE_ );
430  }
431  return UNASSIGNED_SYSTEM_ADDRESS;
432  }

464 bool TCPInterface::SendList( const char **data, const unsigned int *lengths, const int numParameters, const SystemAddress &systemAddress, bool broadcast )
465 {
466  if (isStarted.GetValue()==0)
467  return false;
468  if (data==0)
469  return false;
470  if (systemAddress==UNASSIGNED_SYSTEM_ADDRESS && broadcast==false)
471  return false;
472  unsigned int totalLength=0;
473  int i;
474  for (i=0; i < numParameters; i++)
475  {
476  if (lengths[i]>0)
477  totalLength+=lengths[i];
478  }
479  if (totalLength==0)
480  return false;
481 
482  if (broadcast)
483  {
484  // Send to all, possible exception system
485  for (i=0; i < remoteClientsLength; i++)
486  {
487  if (remoteClients[i].systemAddress!=systemAddress)
488  {
489  remoteClients[i].SendOrBuffer(data, lengths, numParameters);
490  }
491  }
492  }
493  else
494  {
495  // Send to this player
496  if (systemAddress.systemIndex<remoteClientsLength &&
497  remoteClients[systemAddress.systemIndex].systemAddress==systemAddress)
498  {
499  remoteClients[systemAddress.systemIndex].SendOrBuffer(data, lengths, numParameters);
500  }
501  else
502  {
503  for (i=0; i < remoteClientsLength; i++)
504  {
505  if (remoteClients[i].systemAddress==systemAddress )
506  {
507  remoteClients[i].SendOrBuffer(data, lengths, numParameters);
508  }
509  }
510  }
511  }
512 
513 
514  return true;
515 }
