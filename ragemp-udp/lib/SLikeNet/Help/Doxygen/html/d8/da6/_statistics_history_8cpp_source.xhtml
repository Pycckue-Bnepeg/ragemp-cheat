123 {

134  else

138 
139  TimeAndValue tav;

145  queue->recentSumOfSquares -= tav.val * tav.val;

148  }
149  else

152  tav.time=curTime;
153  }
154 
155  tav.val+=val;
156  queue->values.Push(tav, _FILE_AND_LINE_);
157 
158  queue->recentSum += tav.val;
159  queue->recentSumOfSquares += tav.val * tav.val;
160  queue->longTermSum += tav.val;
161  queue->longTermCount = queue->longTermCount + 1;
162  if (queue->longTermLowest > tav.val)
163  queue->longTermLowest = tav.val;

183 bool StatisticsHistory::GetHistorySorted(uint64_t objectId, SHSortOperation sortType, DataStructures::List<StatisticsHistory::TimeAndValueQueue *> &values) const
184 {
185  unsigned int idx = GetObjectIndex(objectId);
186  if (idx == (unsigned int) -1)
187  return false;
188  TrackedObject *to = objects[idx];
189  DataStructures::List<TimeAndValueQueue*> itemList;
190  DataStructures::List<RakString> keyList;
191  to->dataQueues.GetAsList(itemList,keyList,_FILE_AND_LINE_);
192  Time curTime = GetTime();
193 
194  DataStructures::OrderedList<TimeAndValueQueue*, TimeAndValueQueue*,TimeAndValueQueueCompAsc> sortedQueues;
195  for (unsigned int i=0; i < itemList.Size(); i++)
196  {
197  TimeAndValueQueue *tavq = itemList[i];
198  tavq->CullExpiredValues(curTime);
199 
200  if (sortType == SH_SORT_BY_RECENT_SUM_ASCENDING || sortType == SH_SORT_BY_RECENT_SUM_DESCENDING)
201  tavq->sortValue = tavq->GetRecentSum();
202  else if (sortType == SH_SORT_BY_LONG_TERM_SUM_ASCENDING || sortType == SH_SORT_BY_LONG_TERM_SUM_DESCENDING)
203  tavq->sortValue = tavq->GetLongTermSum();
204  else if (sortType == SH_SORT_BY_RECENT_SUM_OF_SQUARES_ASCENDING || sortType == SH_SORT_BY_RECENT_SUM_OF_SQUARES_DESCENDING)
205  tavq->sortValue = tavq->GetRecentSumOfSquares();
206  else if (sortType == SH_SORT_BY_RECENT_AVERAGE_ASCENDING || sortType == SH_SORT_BY_RECENT_AVERAGE_DESCENDING)
207  tavq->sortValue = tavq->GetRecentAverage();
208  else if (sortType == SH_SORT_BY_LONG_TERM_AVERAGE_ASCENDING || sortType == SH_SORT_BY_LONG_TERM_AVERAGE_DESCENDING)
209  tavq->sortValue = tavq->GetLongTermAverage();
210  else if (sortType == SH_SORT_BY_RECENT_HIGHEST_ASCENDING || sortType == SH_SORT_BY_RECENT_HIGHEST_DESCENDING)
211  tavq->sortValue = tavq->GetRecentHighest();
212  else if (sortType == SH_SORT_BY_RECENT_LOWEST_ASCENDING || sortType == SH_SORT_BY_RECENT_LOWEST_DESCENDING)
213  tavq->sortValue = tavq->GetRecentLowest();
214  else if (sortType == SH_SORT_BY_LONG_TERM_HIGHEST_ASCENDING || sortType == SH_SORT_BY_LONG_TERM_HIGHEST_DESCENDING)
215  tavq->sortValue = tavq->GetLongTermHighest();
216  else
217  tavq->sortValue = tavq->GetLongTermLowest();
218 
219  if (
220  sortType == SH_SORT_BY_RECENT_SUM_ASCENDING ||
221  sortType == SH_SORT_BY_LONG_TERM_SUM_ASCENDING ||
222  sortType == SH_SORT_BY_RECENT_SUM_OF_SQUARES_ASCENDING ||
223  sortType == SH_SORT_BY_RECENT_AVERAGE_ASCENDING ||
224  sortType == SH_SORT_BY_LONG_TERM_AVERAGE_ASCENDING ||
225  sortType == SH_SORT_BY_RECENT_HIGHEST_ASCENDING ||
226  sortType == SH_SORT_BY_RECENT_LOWEST_ASCENDING ||
227  sortType == SH_SORT_BY_LONG_TERM_HIGHEST_ASCENDING ||
228  sortType == SH_SORT_BY_LONG_TERM_LOWEST_ASCENDING)
229  sortedQueues.Insert(tavq, tavq, false, _FILE_AND_LINE_, TimeAndValueQueueCompAsc);
230  else
231  sortedQueues.Insert(tavq, tavq, false, _FILE_AND_LINE_, TimeAndValueQueueCompDesc);
232  }
233 
234  for (unsigned int i=0; i < sortedQueues.Size(); i++)
235  values.Push(sortedQueues[i], _FILE_AND_LINE_);
236  return true;

358 void StatisticsHistory::TimeAndValueQueue::MergeSets( const TimeAndValueQueue *lhs, SHDataCategory lhsDataCategory, const TimeAndValueQueue *rhs, SHDataCategory rhsDataCategory, TimeAndValueQueue *output )
359 {
360  // Two ways to merge:
361  // 1. Treat rhs as just more data points.
362  // 1A. Sums are just added. If two values have the same time, just put in queue twice
363  // 1B. longTermLowest and longTermHighest are the lowest and highest of the two sets
364  //
365  // 2. Add by time. If time for the other set is missing, calculate slope to extrapolate
366  // 2A. Have to recalculate recentSum, recentSumOfSquares.
367  // 2B. longTermSum, longTermCount, longTermLowest, longTermHighest are unknown
368 
369  if (lhs!=output)
370  {
371  output->key = lhs->key;
372  output->timeToTrackValues = lhs->timeToTrackValues;
373  }
374  else
375  {
376  output->key = rhs->key;
377  output->timeToTrackValues = rhs->timeToTrackValues;
378  }
379 
380  unsigned int lhsIndex, rhsIndex;
381  lhsIndex=0;
382  rhsIndex=0;
383 
384  // I use local valuesOutput in case lhs==output || rhs==output
385  DataStructures::Queue<TimeAndValue> valuesOutput;
386 
387  if (lhsDataCategory==StatisticsHistory::DC_DISCRETE && rhsDataCategory==StatisticsHistory::DC_DISCRETE)
388  {
389  while (rhsIndex < rhs->values.Size() && lhsIndex < lhs->values.Size())
390  {
391  if (rhs->values[rhsIndex].time < lhs->values[lhsIndex].time)
392  {
393  valuesOutput.Push(rhs->values[rhsIndex], _FILE_AND_LINE_ );
394  rhsIndex++;
395  }
396  else if (rhs->values[rhsIndex].time > lhs->values[lhsIndex].time)
397  {
398  valuesOutput.Push(lhs->values[rhsIndex], _FILE_AND_LINE_ );
399  lhsIndex++;
400  }
401  else
402  {
403  valuesOutput.Push(rhs->values[rhsIndex], _FILE_AND_LINE_ );
404  rhsIndex++;
405  valuesOutput.Push(lhs->values[rhsIndex], _FILE_AND_LINE_ );
406  lhsIndex++;
407  }
408  }
409 
410  while (rhsIndex < rhs->values.Size())
411  {
412  valuesOutput.Push(rhs->values[rhsIndex], _FILE_AND_LINE_ );
413  rhsIndex++;
414  }
415  while (lhsIndex < lhs->values.Size())
416  {
417  valuesOutput.Push(lhs->values[lhsIndex], _FILE_AND_LINE_ );
418  lhsIndex++;
419  }
420 
421  output->recentSum = lhs->recentSum + rhs->recentSum;
422  output->recentSumOfSquares = lhs->recentSumOfSquares + rhs->recentSumOfSquares;
423  output->longTermSum = lhs->longTermSum + rhs->longTermSum;
424  output->longTermCount = lhs->longTermCount + rhs->longTermCount;
425  if (lhs->longTermLowest < rhs->longTermLowest)
426  output->longTermLowest = lhs->longTermLowest;
427  else
428  output->longTermLowest = rhs->longTermLowest;
429  if (lhs->longTermHighest > rhs->longTermHighest)
430  output->longTermHighest = lhs->longTermHighest;
431  else
432  output->longTermHighest = rhs->longTermHighest;
433  }
434  else
435  {
436  TimeAndValue lastTimeAndValueLhs, lastTimeAndValueRhs;
437  lastTimeAndValueLhs.time=0;
438  lastTimeAndValueLhs.val=0;
439  lastTimeAndValueRhs.time=0;
440  lastTimeAndValueRhs.val=0;
441  SHValueType lastSlopeLhs=0;
442  SHValueType lastSlopeRhs=0;
443  Time timeSinceOppositeValue;
444 
445  TimeAndValue newTimeAndValue;
446 
447  while (rhsIndex < rhs->values.Size() && lhsIndex < lhs->values.Size())
448  {
449  if (rhs->values[rhsIndex].time < lhs->values[lhsIndex].time)
450  {
451  timeSinceOppositeValue = rhs->values[rhsIndex].time - lastTimeAndValueLhs.time;
452  newTimeAndValue.val = rhs->values[rhsIndex].val + lastTimeAndValueLhs.val + lastSlopeLhs * timeSinceOppositeValue;
453  newTimeAndValue.time = rhs->values[rhsIndex].time;
454  lastTimeAndValueRhs = rhs->values[rhsIndex];
455  if (rhsIndex>0 && rhs->values[rhsIndex].time != rhs->values[rhsIndex-1].time && rhsDataCategory==StatisticsHistory::DC_CONTINUOUS)
456  lastSlopeRhs = (rhs->values[rhsIndex].val - rhs->values[rhsIndex-1].val) / (SHValueType) (rhs->values[rhsIndex].time - rhs->values[rhsIndex-1].time);
457  rhsIndex++;
458  }
459  else if (lhs->values[lhsIndex].time < rhs->values[rhsIndex].time)
460  {
461  timeSinceOppositeValue = lhs->values[lhsIndex].time - lastTimeAndValueRhs.time;
462  newTimeAndValue.val = lhs->values[lhsIndex].val + lastTimeAndValueRhs.val + lastSlopeRhs * timeSinceOppositeValue;
463  newTimeAndValue.time = lhs->values[lhsIndex].time;
464  lastTimeAndValueLhs = lhs->values[lhsIndex];
465  if (lhsIndex>0 && lhs->values[lhsIndex].time != lhs->values[lhsIndex-1].time && lhsDataCategory==StatisticsHistory::DC_CONTINUOUS)
466  lastSlopeLhs = (lhs->values[lhsIndex].val - lhs->values[lhsIndex-1].val) / (SHValueType) (lhs->values[lhsIndex].time - lhs->values[lhsIndex-1].time);
467  lhsIndex++;
468  }
469  else
470  {
471  newTimeAndValue.val = lhs->values[lhsIndex].val + rhs->values[rhsIndex].val;
472  newTimeAndValue.time = lhs->values[lhsIndex].time;
473  lastTimeAndValueRhs = rhs->values[rhsIndex];
474  lastTimeAndValueLhs = lhs->values[lhsIndex];
475  if (rhsIndex>0 && rhs->values[rhsIndex].time != rhs->values[rhsIndex-1].time && rhsDataCategory==StatisticsHistory::DC_CONTINUOUS)
476  lastSlopeRhs = (rhs->values[rhsIndex].val - rhs->values[rhsIndex-1].val) / (SHValueType) (rhs->values[rhsIndex].time - rhs->values[rhsIndex-1].time);
477  if (lhsIndex>0 && lhs->values[lhsIndex].time != lhs->values[lhsIndex-1].time && lhsDataCategory==StatisticsHistory::DC_CONTINUOUS)
478  lastSlopeLhs = (lhs->values[lhsIndex].val - lhs->values[lhsIndex-1].val) / (SHValueType) (lhs->values[lhsIndex].time - lhs->values[lhsIndex-1].time);
479  lhsIndex++;
480  rhsIndex++;
481  }
482 
483  valuesOutput.Push(newTimeAndValue, _FILE_AND_LINE_ );
484  }
485 
486  while (rhsIndex < rhs->values.Size())
487  {
488  timeSinceOppositeValue = rhs->values[rhsIndex].time - lastTimeAndValueLhs.time;
489  newTimeAndValue.val = rhs->values[rhsIndex].val + lastTimeAndValueLhs.val + lastSlopeLhs * timeSinceOppositeValue;
490  newTimeAndValue.time = rhs->values[rhsIndex].time;
491  valuesOutput.Push(newTimeAndValue, _FILE_AND_LINE_ );
492  rhsIndex++;
493  }
494  while (lhsIndex < lhs->values.Size())
495  {
496  timeSinceOppositeValue = lhs->values[lhsIndex].time - lastTimeAndValueRhs.time;
497  newTimeAndValue.val = lhs->values[lhsIndex].val + lastTimeAndValueRhs.val + lastSlopeRhs * timeSinceOppositeValue;
498  newTimeAndValue.time = lhs->values[lhsIndex].time;
499  valuesOutput.Push(newTimeAndValue, _FILE_AND_LINE_ );
500  lhsIndex++;
501  }
502 
503  output->recentSum = 0;
504  output->recentSumOfSquares = 0;
505  for (unsigned int i=0; i < valuesOutput.Size(); i++)
506  {
507  output->recentSum += valuesOutput[i].val;
508  output->recentSumOfSquares += valuesOutput[i].val * valuesOutput[i].val;
509  }
510  }
511 
512  output->values = valuesOutput;

515 {
516  histogram.Clear(_FILE_AND_LINE_);
517  if (maxSamples==0)
518  return;
519  Time timeRange = GetTimeRange();
520  if (timeRange==0)
521  return;
522  if (maxSamples==1)
523  {
524  StatisticsHistory::TimeAndValue tav;
525  tav.time = timeRange;
526  tav.val = GetRecentSum();
527  histogram.Push(tav, _FILE_AND_LINE_);
528  return;
529  }
530  Time interval = timeRange / maxSamples;
531  if (interval==0)
532  interval=1;
533  unsigned int dataIndex;
534  Time timeBoundary;
535  StatisticsHistory::TimeAndValue currentSum;
536  Time currentTime;
537  SHValueType numSamples;
538  Time endTime;
539 
540  numSamples=0;
541  endTime = values[values.Size()-1].time;
542  dataIndex=0;
543  currentTime=values[0].time;
544  currentSum.val=0;
545  currentSum.time=values[0].time + interval / 2;
546  timeBoundary = values[0].time + interval;
547  while (timeBoundary <= endTime)
548  {
549  while (dataIndex < values.Size() && values[dataIndex].time <= timeBoundary)
550  {
551  currentSum.val += values[dataIndex].val;
552  dataIndex++;
553  numSamples++;
554  }
555 
556  if (dataCategory==DC_CONTINUOUS)
557  {
558  if (dataIndex > 0 &&
559  dataIndex < values.Size() &&
560  values[dataIndex-1].time < timeBoundary &&
561  values[dataIndex].time > timeBoundary)
562  {
563  SHValueType interpolatedValue = Interpolate(values[dataIndex-1], values[dataIndex], timeBoundary);
564  currentSum.val+=interpolatedValue;
565  numSamples++;
566  }
567 
568  if (numSamples > 1)
569  {
570  currentSum.val /= numSamples;
571  }
572  }
573 
574  histogram.Push(currentSum, _FILE_AND_LINE_);
575  currentSum.time=timeBoundary + interval / 2;
576  timeBoundary += interval;
577  currentSum.val=0;
578  numSamples=0;
579  }
580 
581 
582  if ( timeClipStart!=0 && histogram.Size()>=1)
583  {
584  timeClipStart = histogram.Peek().time+timeClipStart;
585  if (histogram.PeekTail().time < timeClipStart)
586  {
587  histogram.Clear(_FILE_AND_LINE_);
588  }
589  else if (histogram.Size()>=2 && histogram.Peek().time < timeClipStart)
590  {
591  StatisticsHistory::TimeAndValue tav;
592 
593  do
594  {
595  tav = histogram.Pop();
596 
597  if (histogram.Peek().time == timeClipStart)
598  {
599  break;
600  }
601  else if (histogram.Peek().time > timeClipStart)
602  {
603  StatisticsHistory::TimeAndValue tav2;
604  tav2.val = StatisticsHistory::TimeAndValueQueue::Interpolate(tav, histogram.Peek(), timeClipStart);
605  tav2.time=timeClipStart;
606  histogram.PushAtHead(tav2, 0, _FILE_AND_LINE_);
607  break;
608  }
609  } while (histogram.Size()>=2);
610  }
611  }
612 
613  if ( timeClipEnd!=0 && histogram.Size()>=1)
614  {
615  timeClipEnd = histogram.PeekTail().time-timeClipEnd;
616  if (histogram.Peek().time > timeClipEnd)
617  {
618  histogram.Clear(_FILE_AND_LINE_);
619  }
620  else if (histogram.Size()>=2 && histogram.PeekTail().time > timeClipEnd)
621  {
622  StatisticsHistory::TimeAndValue tav;
623 
624  do
625  {
626  tav = histogram.PopTail();
627 
628  if (histogram.PeekTail().time == timeClipEnd)
629  {
630  break;
631  }
632  else if (histogram.PeekTail().time < timeClipEnd)
633  {
634  StatisticsHistory::TimeAndValue tav2;
635  tav2.val = StatisticsHistory::TimeAndValueQueue::Interpolate(tav, histogram.PeekTail(), timeClipEnd);
636  tav2.time=timeClipEnd;
637  histogram.Push(tav2, _FILE_AND_LINE_);
638  break;
639  }
640  } while (histogram.Size()>=2);
641  }
