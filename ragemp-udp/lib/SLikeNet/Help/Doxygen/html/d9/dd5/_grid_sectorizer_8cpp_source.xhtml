30 void GridSectorizer::Init(const float _maxCellWidth, const float _maxCellHeight, const float minX, const float minY, const float maxX, const float maxY)
31 {
32  RakAssert(_maxCellWidth > 0.0f && _maxCellHeight > 0.0f);
33  if (grid)
34  SLNet::OP_DELETE_ARRAY(grid, _FILE_AND_LINE_);
35 
36  cellOriginX=minX;
37  cellOriginY=minY;
38  gridWidth=maxX-minX;
39  gridHeight=maxY-minY;
40  gridCellWidthCount=(int) ceil(gridWidth/_maxCellWidth);
41  gridCellHeightCount=(int) ceil(gridHeight/_maxCellHeight);
42  // Make the cells slightly smaller, so we allocate an extra unneeded cell if on the edge. This way we don't go outside the array on rounding errors.
43  cellWidth=gridWidth/gridCellWidthCount;
44  cellHeight=gridHeight/gridCellHeightCount;
45  invCellWidth = 1.0f / cellWidth;
46  invCellHeight = 1.0f / cellHeight;
47 
48 #ifdef _USE_ORDERED_LIST
49  grid = SLNet::OP_NEW<DataStructures::OrderedList<void*, void*>>(gridCellWidthCount*gridCellHeightCount, _FILE_AND_LINE_ );
50  DataStructures::OrderedList<void*,void*>::IMPLEMENT_DEFAULT_COMPARISON();
51 #else
52  grid = SLNet::OP_NEW_ARRAY<DataStructures::List<void*> >(gridCellWidthCount*gridCellHeightCount, _FILE_AND_LINE_ );
53 #endif
54 }

98 void GridSectorizer::MoveEntry(void *entry, const float sourceMinX, const float sourceMinY, const float sourceMaxX, const float sourceMaxY,
99  const float destMinX, const float destMinY, const float destMaxX, const float destMaxY)
100 {
101  RakAssert(cellWidth>0.0f);
102  RakAssert(sourceMinX < sourceMaxX && sourceMinY < sourceMaxY);
103  RakAssert(destMinX < destMaxX && destMinY < destMaxY);
104 
105  if (PositionCrossesCells(sourceMinX, sourceMinY, destMinX, destMinY)==false &&
106  PositionCrossesCells(destMinX, destMinY, destMinX, destMinY)==false)
107  return;
108 
109  int xStartSource, yStartSource, xEndSource, yEndSource;
110  int xStartDest, yStartDest, xEndDest, yEndDest;
111  int xCur, yCur;
112  xStartSource=WorldToCellXOffsetAndClamped(sourceMinX);
113  yStartSource=WorldToCellYOffsetAndClamped(sourceMinY);
114  xEndSource=WorldToCellXOffsetAndClamped(sourceMaxX);
115  yEndSource=WorldToCellYOffsetAndClamped(sourceMaxY);
116 
117  xStartDest=WorldToCellXOffsetAndClamped(destMinX);
118  yStartDest=WorldToCellYOffsetAndClamped(destMinY);
119  xEndDest=WorldToCellXOffsetAndClamped(destMaxX);
120  yEndDest=WorldToCellYOffsetAndClamped(destMaxY);
121 
122  // Remove source that is not in dest
123  for (xCur=xStartSource; xCur <= xEndSource; ++xCur)
124  {
125  for (yCur=yStartSource; yCur <= yEndSource; ++yCur)
126  {
127  if (xCur < xStartDest || xCur > xEndDest ||
128  yCur < yStartDest || yCur > yEndDest)
129  {
130  grid[yCur*gridCellWidthCount+xCur].RemoveIfExists(entry);
131  }
132  }
133  }
134 
135  // Add dest that is not in source
136  for (xCur=xStartDest; xCur <= xEndDest; ++xCur)
137  {
138  for (yCur=yStartDest; yCur <= yEndDest; ++yCur)
139  {
140  if (xCur < xStartSource || xCur > xEndSource ||
141  yCur < yStartSource || yCur > yEndSource)
142  {
143  grid[yCur*gridCellWidthCount+xCur].Insert(entry,entry, true);

149 void GridSectorizer::GetEntries(DataStructures::List<void*>& intersectionList, const float minX, const float minY, const float maxX, const float maxY)
150 {
151 #ifdef _USE_ORDERED_LIST
152  DataStructures::OrderedList<void*, void*>* cell;
153 #else
154  DataStructures::List<void*>* cell;
155 #endif
156  int xStart, yStart, xEnd, yEnd, xCur, yCur;
157  unsigned index;
158  xStart=WorldToCellXOffsetAndClamped(minX);
159  yStart=WorldToCellYOffsetAndClamped(minY);
160  xEnd=WorldToCellXOffsetAndClamped(maxX);
161  yEnd=WorldToCellYOffsetAndClamped(maxY);
162 
163  intersectionList.Clear(true, _FILE_AND_LINE_);
164  for (xCur=xStart; xCur <= xEnd; ++xCur)
165  {
166  for (yCur=yStart; yCur <= yEnd; ++yCur)
167  {
168  cell = grid+yCur*gridCellWidthCount+xCur;
