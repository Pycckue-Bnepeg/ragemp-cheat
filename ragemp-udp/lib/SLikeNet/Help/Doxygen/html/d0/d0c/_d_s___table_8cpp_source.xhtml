616 void Table::QueryTable(unsigned *columnIndicesSubset, unsigned numColumnSubset, FilterQuery *inclusionFilters, unsigned numInclusionFilters, unsigned *rowIds, unsigned numRowIDs, Table *result)
617 {
618  unsigned i;
619  DataStructures::List<unsigned> columnIndicesToReturn;
620 
621  // Clear the result table.
622  result->Clear();
623 
624  if (columnIndicesSubset && numColumnSubset>0)
625  {
626  for (i=0; i < numColumnSubset; i++)
627  {
628  if (columnIndicesSubset[i]<columns.Size())
629  columnIndicesToReturn.Insert(columnIndicesSubset[i], _FILE_AND_LINE_);
630  }
631  }
632  else
633  {
634  for (i=0; i < columns.Size(); i++)
635  columnIndicesToReturn.Insert(i, _FILE_AND_LINE_);
636  }
637 
638  if (columnIndicesToReturn.Size()==0)
639  return; // No valid columns specified
640 
641  for (i=0; i < columnIndicesToReturn.Size(); i++)
642  {
643  result->AddColumn(columns[columnIndicesToReturn[i]].columnName,columns[columnIndicesToReturn[i]].columnType);
644  }
645 
646  // Get the column indices of the filter queries.
647  DataStructures::List<unsigned> inclusionFilterColumnIndices;
648  if (inclusionFilters && numInclusionFilters>0)
649  {
650  for (i=0; i < numInclusionFilters; i++)
651  {
652  if (inclusionFilters[i].columnName[0])
653  inclusionFilters[i].columnIndex=ColumnIndex(inclusionFilters[i].columnName);
654  if (inclusionFilters[i].columnIndex<columns.Size())
655  inclusionFilterColumnIndices.Insert(inclusionFilters[i].columnIndex, _FILE_AND_LINE_);
656  else
657  inclusionFilterColumnIndices.Insert((unsigned)-1, _FILE_AND_LINE_);
658  }
659  }
660 
661  if (rowIds==0 || numRowIDs==0)
662  {
663  // All rows
664  DataStructures::Page<unsigned, Row*, _TABLE_BPLUS_TREE_ORDER> *cur = rows.GetListHead();
665  while (cur)
666  {
667  for (i=0; i < (unsigned)cur->size; i++)
668  {
669  QueryRow(inclusionFilterColumnIndices, columnIndicesToReturn, cur->keys[i], cur->data[i], inclusionFilters, result);
670  }
671  cur=cur->next;
672  }
673  }
674  else
675  {
676  // Specific rows
677  Row *row;
678  for (i=0; i < numRowIDs; i++)
679  {
680  if (rows.Get(rowIds[i], row))
681  {
682  QueryRow(inclusionFilterColumnIndices, columnIndicesToReturn, rowIds[i], row, inclusionFilters, result);
683  }
684  }
685  }

688 void Table::QueryRow(DataStructures::List<unsigned> &inclusionFilterColumnIndices, DataStructures::List<unsigned> &columnIndicesToReturn, unsigned key, Table::Row* row, FilterQuery *inclusionFilters, Table *result)
689 {
690  bool pass=false;
691  unsigned columnIndex;
692  unsigned j;
693 
694  // If no inclusion filters, just add the row
695  if (inclusionFilterColumnIndices.Size()==0)
696  {
697  result->AddRowColumns(key, row, columnIndicesToReturn);
698  }
699  else
700  {
701  // Go through all inclusion filters. Only add this row if all filters pass.
702  for (j=0; j<inclusionFilterColumnIndices.Size(); j++)
703  {
704  columnIndex=inclusionFilterColumnIndices[j];
705  if (columnIndex!=(unsigned)-1 && row->cells[columnIndex]->isEmpty==false )
706  {
707  if (columns[inclusionFilterColumnIndices[j]].columnType==STRING &&
708  (row->cells[columnIndex]->c==0 ||
709  inclusionFilters[j].cellValue->c==0) )
710  continue;
711 
712  switch (inclusionFilters[j].operation)
713  {
714  case QF_EQUAL:
715  switch(columns[inclusionFilterColumnIndices[j]].columnType)
716  {
717  case NUMERIC:
718  pass=row->cells[columnIndex]->i==inclusionFilters[j].cellValue->i;
719  break;
720  case STRING:
721  pass=strcmp(row->cells[columnIndex]->c,inclusionFilters[j].cellValue->c)==0;
722  break;
723  case BINARY:
724  pass=row->cells[columnIndex]->i==inclusionFilters[j].cellValue->i &&
725  memcmp(row->cells[columnIndex]->c,inclusionFilters[j].cellValue->c, (int) row->cells[columnIndex]->i)==0;
726  break;
727  case POINTER:

752  {
753  case NUMERIC:

768  {
769  case NUMERIC:

771  break;
772  case STRING:

777  case POINTER:

793  case POINTER:

795  break;
796  }
797  break;
798  case QF_LESS_THAN_EQ:
799  switch(columns[inclusionFilterColumnIndices[j]].columnType)
800  {
801  case NUMERIC:
802  pass=row->cells[columnIndex]->i<=inclusionFilters[j].cellValue->i;
803  break;
804  case STRING:
805  pass=strcmp(row->cells[columnIndex]->c,inclusionFilters[j].cellValue->c)<=0;
806  break;
807  case BINARY:
808  break;
809  case POINTER:
810  pass=row->cells[columnIndex]->ptr<=inclusionFilters[j].cellValue->ptr;
811  break;
812  }
813  break;
814  case QF_IS_EMPTY:
815  pass=false;
816  break;
817  case QF_NOT_EMPTY:
818  pass=true;
819  break;
820  default:
821  pass=false;
822  RakAssert(0);
823  break;
824  }

832  }

834  if (pass==false)
835  break;
836  }
837 
838  if (pass)
839  {
