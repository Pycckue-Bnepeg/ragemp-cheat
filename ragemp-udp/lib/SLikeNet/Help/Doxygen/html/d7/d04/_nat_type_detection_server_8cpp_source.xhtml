377 void NatTypeDetectionServer::OnDetectionRequest(Packet *packet)
378 {
379  unsigned int i = GetDetectionAttemptIndex(packet->systemAddress);
380 
381  SLNet::BitStream bsIn(packet->data, packet->length, false);
382  bsIn.IgnoreBytes(1);
383  bool isRequest=false;
384  bsIn.Read(isRequest);
385  if (isRequest)
386  {
387  if (i!=(unsigned int)-1)
388  return; // Already in progress
389 
390  NATDetectionAttempt nda;
391  nda.detectionState=STATE_NONE;
392  nda.systemAddress=packet->systemAddress;
393  nda.guid=packet->guid;
394  bsIn.Read(nda.c2Port);
395  nda.nextStateTime=0;
396  nda.timeBetweenAttempts=rakPeerInterface->GetLastPing(nda.systemAddress)*3+50;
397  natDetectionAttempts.Push(nda, _FILE_AND_LINE_);
398  }
399  else
400  {
401  if (i==(unsigned int)-1)
402  return; // Unknown
403  // They are done
404  natDetectionAttempts.RemoveAtIndexFast(i);
405  }
