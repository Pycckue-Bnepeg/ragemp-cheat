354 void NatPunchthroughClient::OnPunchthroughFailure(void)
355 {
356  if (pc.retryOnFailure==false)
357  {
358  if (natPunchthroughDebugInterface)
359  {
360  char ipAddressString[32];
361  sp.targetAddress.ToString(true, ipAddressString, static_cast<size_t>(32));
362  char guidString[128];
363  sp.targetGuid.ToString(guidString, 128);
364  natPunchthroughDebugInterface->OnClientMessage(SLNet::RakString("Failed punchthrough once. Returning failure to guid %s, system address %s to user.", guidString, ipAddressString));
365  }
366 
367  PushFailure();
368  OnReadyForNextPunchthrough();
369  return;
370  }
371 
372  unsigned int i;
373  for (i=0; i < failedAttemptList.Size(); i++)
374  {
375  if (failedAttemptList[i].guid==sp.targetGuid)
376  {
377  if (natPunchthroughDebugInterface)
378  {
379  char ipAddressString[32];
380  sp.targetAddress.ToString(true, ipAddressString, static_cast<size_t>(32));
381  char guidString[128];
382  sp.targetGuid.ToString(guidString, 128);
383  natPunchthroughDebugInterface->OnClientMessage(SLNet::RakString("Failed punchthrough twice. Returning failure to guid %s, system address %s to user.", guidString, ipAddressString));
384  }
385 
386  // Failed a second time, so return failed to user
387  PushFailure();
388 
389  OnReadyForNextPunchthrough();
390 
391  failedAttemptList.RemoveAtIndexFast(i);
392  return;
393  }
394  }
395 
396  if (rakPeerInterface->GetConnectionState(sp.facilitator)!=IS_CONNECTED)
397  {
398  if (natPunchthroughDebugInterface)
399  {
400  char ipAddressString[32];
401  sp.targetAddress.ToString(true, ipAddressString, static_cast<size_t>(32));
402  char guidString[128];
403  sp.targetGuid.ToString(guidString, 128);
404  natPunchthroughDebugInterface->OnClientMessage(SLNet::RakString("Not connected to facilitator, so cannot retry punchthrough after first failure. Returning failure onj guid %s, system address %s to user.", guidString, ipAddressString));
405  }
406 
407  // Failed, and can't try again because no facilitator
408  PushFailure();
409  return;
410  }
411 
412  if (natPunchthroughDebugInterface)
413  {
414  char ipAddressString[32];
415  sp.targetAddress.ToString(true, ipAddressString, static_cast<size_t>(32));
416  char guidString[128];
417  sp.targetGuid.ToString(guidString, 128);
418  natPunchthroughDebugInterface->OnClientMessage(SLNet::RakString("First punchthrough failure on guid %s, system address %s. Reattempting.", guidString, ipAddressString));
419  }
420 
421  // Failed the first time. Add to the failure queue and try again
422  AddrAndGuid aag;
423  aag.addr=sp.targetAddress;
424  aag.guid=sp.targetGuid;
425  failedAttemptList.Push(aag, _FILE_AND_LINE_);
426 
427  // Tell the server we are ready
428  OnReadyForNextPunchthrough();
429 
430  // If we are the sender, try again, immediately if possible, else added to the queue on the faciltiator
431  if (sp.weAreSender)
432  SendPunchthrough(sp.targetGuid, sp.facilitator);

434 PluginReceiveResult NatPunchthroughClient::OnReceive(Packet *packet)
435 {
436  switch (packet->data[0])
437  {
438  case ID_NAT_GET_MOST_RECENT_PORT:
439  {
440  OnGetMostRecentPort(packet);
441  return RR_STOP_PROCESSING_AND_DEALLOCATE;
442  }
443  case ID_NAT_PUNCHTHROUGH_FAILED:
444  case ID_NAT_PUNCHTHROUGH_SUCCEEDED:
445  if (packet->wasGeneratedLocally==false)
446  return RR_STOP_PROCESSING_AND_DEALLOCATE;
447  break;
448  case ID_NAT_RESPOND_BOUND_ADDRESSES:
449  {
450  SLNet::BitStream bs(packet->data,packet->length,false);
451  bs.IgnoreBytes(sizeof(MessageID));
452  unsigned char boundAddressCount;
453  bs.Read(boundAddressCount);
454  if (boundAddressCount<2)
455  {
456  if (natPunchthroughDebugInterface)
457  natPunchthroughDebugInterface->OnClientMessage(RakString("INCAPABLE_PORT_STRIDE. My external ID is %s", rakPeerInterface->GetExternalID(packet->systemAddress).ToString()));
458 
459  hasPortStride=INCAPABLE_PORT_STRIDE;
460  SendQueuedOpenNAT();
461  }
462  SystemAddress boundAddresses[MAXIMUM_NUMBER_OF_INTERNAL_IDS];
463  for (int i=0; i < boundAddressCount && i < MAXIMUM_NUMBER_OF_INTERNAL_IDS; i++)
464  {
465  bs.Read(boundAddresses[i]);
466  if (boundAddresses[i]!=packet->systemAddress)
467  {
468  SLNet::BitStream outgoingBs;
469  outgoingBs.Write((MessageID)ID_NAT_PING);
470  uint16_t externalPort = rakPeerInterface->GetExternalID(packet->systemAddress).GetPort();
471  outgoingBs.Write( externalPort );
472  rakPeerInterface->SendOutOfBand((const char*) boundAddresses[i].ToString(false),boundAddresses[i].GetPort(),(const char*) outgoingBs.GetData(),outgoingBs.GetNumberOfBytesUsed());
473  break;
474  }
475  }
476  }
477  return RR_STOP_PROCESSING_AND_DEALLOCATE;
478  case ID_OUT_OF_BAND_INTERNAL:
479  if (packet->length>=2 && packet->data[1]==ID_NAT_PONG)
480  {
481  SLNet::BitStream bs(packet->data,packet->length,false);
482  bs.IgnoreBytes(sizeof(MessageID)*2);
483  uint16_t externalPort;
484  bs.Read(externalPort);
485  uint16_t externalPort2;
486  bs.Read(externalPort2);
487  portStride = externalPort2 - externalPort;
488  mostRecentExternalPort = externalPort2;
489  hasPortStride=HAS_PORT_STRIDE;
490 
491  if (natPunchthroughDebugInterface)
492  natPunchthroughDebugInterface->OnClientMessage(RakString("HAS_PORT_STRIDE %i. First external port %i. Second external port %i.", portStride, externalPort, externalPort2));
493 
494  SendQueuedOpenNAT();
495  return RR_STOP_PROCESSING_AND_DEALLOCATE;
496  }
497  else if (packet->length>=2 &&
498  (packet->data[1]==ID_NAT_ESTABLISH_UNIDIRECTIONAL || packet->data[1]==ID_NAT_ESTABLISH_BIDIRECTIONAL) &&
499  sp.nextActionTime!=0)
500  {
501  SLNet::BitStream bs(packet->data,packet->length,false);
502  bs.IgnoreBytes(2);
503  uint16_t sessionId;
504  bs.Read(sessionId);
505 // RakAssert(sessionId<100);
506  if (sessionId!=sp.sessionId)
507  break;
508 
509  char ipAddressString[32];
510  packet->systemAddress.ToString(true,ipAddressString,static_cast<size_t>(32));
511  // sp.targetGuid==packet->guid is because the internal IP addresses reported may include loopbacks not reported by RakPeer::IsLocalIP()
512  if (packet->data[1]==ID_NAT_ESTABLISH_UNIDIRECTIONAL && sp.targetGuid==packet->guid)
513  {
514 
515  if (sp.testMode!=SendPing::PUNCHING_FIXED_PORT)
516  {
517  sp.testMode=SendPing::PUNCHING_FIXED_PORT;
518  sp.retryCount+=sp.attemptCount*pc.UDP_SENDS_PER_PORT_EXTERNAL;
519  sp.targetAddress=packet->systemAddress;
520  // Keeps trying until the other side gives up too, in case it is unidirectional
521  sp.punchingFixedPortAttempts=pc.UDP_SENDS_PER_PORT_EXTERNAL*(pc.MAX_PREDICTIVE_PORT_RANGE+1);
522 
523  if (natPunchthroughDebugInterface)
524  {
525  char guidString[128];
526  sp.targetGuid.ToString(guidString, 128);
527  natPunchthroughDebugInterface->OnClientMessage(SLNet::RakString("PUNCHING_FIXED_PORT: Received ID_NAT_ESTABLISH_UNIDIRECTIONAL from guid %s, system address %s.", guidString, ipAddressString));
528  }
529  }
530  else {
531  if (natPunchthroughDebugInterface)
532  {
533  char guidString[128];
534  sp.targetGuid.ToString(guidString, 128);
535  natPunchthroughDebugInterface->OnClientMessage(SLNet::RakString("Received ID_NAT_ESTABLISH_UNIDIRECTIONAL from guid %s, system address %s.", guidString, ipAddressString));
536  }
537  }
538 
539  SendOutOfBand(sp.targetAddress,ID_NAT_ESTABLISH_BIDIRECTIONAL);
540  }
541  else if (packet->data[1]==ID_NAT_ESTABLISH_BIDIRECTIONAL &&
542  sp.targetGuid==packet->guid)
543  {
544  // They send back our port
545  unsigned short ourExternalPort;
546  bs.Read(ourExternalPort);
547  if (mostRecentExternalPort==0)
548  {
549  mostRecentExternalPort=ourExternalPort;
550 
551  if (natPunchthroughDebugInterface)
552  {
553  natPunchthroughDebugInterface->OnClientMessage(SLNet::RakString("ID_NAT_ESTABLISH_BIDIRECTIONAL mostRecentExternalPort first time set to %i", mostRecentExternalPort));
554  }
555  }
556  else
557  {
558  if (sp.testMode!=SendPing::TESTING_INTERNAL_IPS && sp.testMode!=SendPing::WAITING_FOR_INTERNAL_IPS_RESPONSE)
559  {
560  if (hasPortStride!=HAS_PORT_STRIDE)
561  {
562  portStride = ourExternalPort - mostRecentExternalPort;
563  hasPortStride=HAS_PORT_STRIDE;
564 
565  if (natPunchthroughDebugInterface)
566  {
567  natPunchthroughDebugInterface->OnClientMessage(RakString("ID_NAT_ESTABLISH_BIDIRECTIONAL: Estimated port stride from incoming connection at %i. ourExternalPort=%i mostRecentExternalPort=%i", portStride, ourExternalPort, mostRecentExternalPort));
568  }
569 
570  SendQueuedOpenNAT();
571  }
572 
573  //nextExternalPort += portStride * (pc.MAX_PREDICTIVE_PORT_RANGE+1);
574  mostRecentExternalPort = ourExternalPort;
575 
576  if (natPunchthroughDebugInterface)
577  {
578  natPunchthroughDebugInterface->OnClientMessage(RakString("ID_NAT_ESTABLISH_BIDIRECTIONAL: New mostRecentExternalPort %i", mostRecentExternalPort));
579  }
580  }
581  }
582  SendOutOfBand(packet->systemAddress,ID_NAT_ESTABLISH_BIDIRECTIONAL);
583 
584  // Tell the user about the success
