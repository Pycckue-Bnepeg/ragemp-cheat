58 uint32_t SuperFastHashIncremental (const char * data, int len, unsigned int lastHash )
59 {
60  uint32_t hash = (uint32_t) lastHash;
61  uint32_t tmp;
62  int rem;
63 
64  if (len <= 0 || data == NULL) return 0;
65 
66  rem = len & 3;
67  len >>= 2;
68 
69  /* Main loop */
70  for (;len > 0; len--) {
71  hash += get16bits (data);
72  tmp = (get16bits (data+2) << 11) ^ hash;
73  hash = (hash << 16) ^ tmp;
74  data += 2*sizeof (uint16_t);
75  hash += hash >> 11;
76  }
77 
78  /* Handle end cases */
79  switch (rem) {
80  case 3: hash += get16bits (data);
81  hash ^= hash << 16;
82  hash ^= data[sizeof (uint16_t)] << 18;
83  hash += hash >> 11;
84  break;
85  case 2: hash += get16bits (data);
86  hash ^= hash << 11;
87  hash += hash >> 17;
88  break;
89  case 1: hash += *data;
90  hash ^= hash << 10;
91  hash += hash >> 1;
92  }
93 
94  /* Force "avalanching" of final 127 bits */
95  hash ^= hash << 3;
96  hash += hash >> 5;
97  hash ^= hash << 4;
98  hash += hash >> 17;
99  hash ^= hash << 25;
100  hash += hash >> 6;
101 
102  return (uint32_t) hash;
103 
104 }
